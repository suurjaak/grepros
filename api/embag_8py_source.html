161  def _populate_message(self, msg, embagval):
162  """Returns the ROS1 message populated from a corresponding embag.RosValue."""
163  for name, typename in rosapi.get_message_fields(msg).items():
164  v, scalarname = embagval.get(name), rosapi.scalar(typename)
165  if typename in rosapi.ROS_BUILTIN_TYPES: # Single built-in type
166  msgv = getattr(embagval, name)
167  elif scalarname in rosapi.ROS_BUILTIN_TYPES: # List of built-in types
168  msgv = list(v)
169  elif typename in rosapi.ROS_TIME_TYPES: # Single temporal type
170  cls = next(k for k, v in rosapi.ROS_TIME_CLASSES.items() if v == typename)
171  msgv = cls(v.secs, v.nsecs)
172  elif scalarname in rosapi.ROS_TIME_TYPES: # List of temporal types
173  cls = next(k for k, v in rosapi.ROS_TIME_CLASSES.items() if v == scalarname)
174  msgv = [cls(x.secs, x.nsecs) for x in v]
175  elif typename == scalarname: # Single subtype
176  msgv = self._populate_message(self.get_message_class(typename)(), v)
177  else: # List of subtypes
178  cls = self.get_message_class(scalarname)
179  msgv = [self._populate_message(cls(), x) for x in v]
180  setattr(msg, name, msgv)
181  return msg
