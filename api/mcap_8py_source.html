150  def _decode_message(self, message, channel, schema):
151  """
152  Returns ROS message deserialized from binary data.
153 
154  @param message mcap.records.Message instance
155  @param channel mcap.records.Channel instance for message
156  @param shcema mcap.records.Schema instance for message type
157  """
158  typekey = (typename, typehash) = self._schemas[schema.id]
159  if ros2 and typekey not in self._types:
160  try: # Try loading class from disk for full compatibility
161  cls = rosapi.get_message_class(typename)
162  clshash = rosapi.get_message_type_hash(cls)
163  if typehash == clshash: self._types[typekey] = cls
164  except Exception: pass # ModuleNotFoundError, AttributeError etc
165  if ros2 and typekey in self._types:
166  msg = ros2.deserialize_message(message.data, self._types[typekey])
167  else:
168  msg = self._decoder.decode(schema=schema, message=message)
169  if ros2: # MCAP ROS2 message classes need monkey-patching with expected API
170  msg = self._patch_message(msg, *self._schemas[schema.id])
171  # Register serialized binary, as MCAP does not support serializing its own creations
172  rosapi.TypeMeta.make(msg, channel.topic, data=message.data)
173  if typekey not in self._types: self._types[typekey] = type(msg)
174  return msg

177  def _patch_message(self, message, typename, typehash):
178  """
179  Patches MCAP ROS2 message with expected attributes and methods, recursively.
180 
181  @param message ROS message instance as returned from mcap_ros2.decoder
182  @param typename ROS message type name, like "std_msgs/Bool"
183  @param typehash ROS message type hash
184  @return original message patched, or new instance if ROS2 temporal type
185  """
186  result = message
187  # [(field value, (field type name, field type hash), parent, (field name, ?array index))]
188  stack = [(message, (typename, typehash), None, ())]
189  while stack:
190  msg, typekey, parent, path = stack.pop(0)
191  mycls, typename = type(msg), typekey[0]
192 
193  if typename in self._temporal_ctors:
194  # Convert temporal types to ROS2 temporal types for grepros to recognize
195  msg2 = self._temporal_ctors[typename](sec=msg.sec, nanosec=msg.nanosec)
196  if msg is message: result = msg2 # Replace input message
197  elif len(path) == 1: setattr(parent, path[0], msg2) # Set scalar field
198  else: getattr(parent, path[0])[path[1]] = msg2 # Set array field element
199  continue # while stack
200 
201  if typekey not in self._typefields:
202  fields = rosapi.parse_definition_fields(typename, self._typedefs[typekey])
203  self._typefields[typekey] = fields
204  self._field_subtypes[typekey] = {n: (s, h) for n, t in fields.items()
205  for s in [rosapi.scalar(t)]
206  if s not in rosapi.ROS_BUILTIN_TYPES
207  for h in [self._type_subtypes[typekey][s]]}
208 
209  # mcap_ros2 creates a dynamic class for each message, having __slots__
210  # but no other ROS2 API hooks; even the class module is "mcap_ros2.dynamic".
211  mycls.__module__ = typename.split("/", maxsplit=1)[0]
212  mycls.SLOT_TYPES = () # rosidl_runtime_py.utilities.is_message() checks for presence
213  mycls._fields_and_field_types = dict(self._typefields[typekey])
214  mycls.get_fields_and_field_types = message_get_fields_and_field_types
215  mycls.__copy__ = copy_with_slots # MCAP message classes lack support for copy
216  mycls.__deepcopy__ = deepcopy_with_slots
217 
218  for name, subtypekey in self._field_subtypes[typekey].items():
219  v = getattr(msg, name)
220  if isinstance(v, list): # Queue each array element for patching
221  stack.extend((x, subtypekey, msg, (name, i)) for i, x in enumerate(v))
222  else: # Queue scalar field for patching
223  stack.append((v, subtypekey, msg, (name, )))
224 
225  return result

228  def _populate_meta(self):
229  """Populates bag metainfo."""
230  summary = self._reader.get_summary()
231  self._start_time = summary.statistics.message_start_time / 1E9
232  self._end_time = summary.statistics.message_end_time / 1E9
233 
234  defhashes = {} # Cached {type definition full text: type hash}
235  for cid, channel in summary.channels.items():
236  schema = summary.schemas[channel.schema_id]
237  topic, typename = channel.topic, schema.name
238  if ros2: typename = ros2.canonical(typename)
239 
240  typedef = schema.data.decode("utf-8") # Full definition including subtype definitions
241  subtypedefs, nesting = rosapi.parse_definition_subtypes(typedef, nesting=True)
242  typehash = channel.metadata.get("md5sum") or \
243  rosapi.calculate_definition_hash(typename, typedef,
244  tuple(subtypedefs.items()))
245  topickey, typekey = (topic, typename, typehash), (typename, typehash)
246 
247  qoses = None
248  if channel.metadata.get("offered_qos_profiles"):
249  try: qoses = yaml.safe_load(channel.metadata["offered_qos_profiles"])
250  except Exception as e:
251  ConsolePrinter.warn("Error parsing topic QoS profiles from %r: %s.",
252  channel.metadata["offered_qos_profiles"], e)
253 
254  self._topics.setdefault(topickey, 0)
255  self._topics[topickey] += summary.statistics.channel_message_counts[cid]
256  self._typedefs[typekey] = typedef
257  defhashes[typedef] = typehash
258  for t, d in subtypedefs.items(): # Populate subtype definitions and hashes
259  if d in defhashes: h = defhashes[d]
260  else: h = rosapi.calculate_definition_hash(t, d, tuple(subtypedefs.items()))
261  self._typedefs.setdefault((t, h), d)
262  self._type_subtypes.setdefault(typekey, {})[t] = h
263  defhashes[d] = h
264  for t, subtypes in nesting.items(): # Populate all nested type references
265  h = self._type_subtypes[typekey][t]
266  for t2 in subtypes:
267  h2 = self._type_subtypes[typekey][t2]
268  self._type_subtypes.setdefault((t, h), {})[t2] = h2
269 
270  if qoses: self._qoses[topickey] = qoses
271  self._schemas[schema.id] = typekey
