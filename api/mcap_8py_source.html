128 """

177 def get_type_and_topic_info(self, topic_filters=None):
178 """
179 Returns thorough metainfo on topic and message types.
180
181 @param topic_filters list of topics or a single topic to filter returned topics-dict by,
182 if any
183 @return TypesAndTopicsTuple(msg_types, topics) namedtuple,
184 msg_types as dict of {typename: typehash},
185 topics as a dict of {topic: TopicTuple() namedtuple}.
186 """
187 topics = topic_filters
188 topics = topics if isinstance(topics, (list, set, tuple)) else [topics] if topics else []
189 if self._ttinfo and (not topics or set(topics) == set(t for t, _, _ in self._topics)):
190 return self._ttinfo
191 if self.closedclosedclosedclosed: raise ValueError("I/O operation on closed file.")
192
193 topics = topic_filters
194 topics = topics if isinstance(topics, (list, set, tuple)) else [topics] if topics else []
195 msgtypes = {n: h for t, n, h in self._topics}
196 topicdict = {}

198 def median(vals):
199 """Returns median value from given sorted numbers."""
200 vlen = len(vals)
201 return None if not vlen else vals[vlen // 2] if vlen % 2 else \
202 float(vals[vlen // 2 - 1] + vals[vlen // 2]) / 2
203
204 for (t, n, _), c in sorted(self._topics.items()):
205 if topics and t not in topics: continue # for
206 mymedian = None
207 if c > 1 and self._reader:
208 stamps = sorted(m.publish_time / 1E9 for _, _, m in self._reader.iter_messages([t]))
209 mymedian = median(sorted(s1 - s0 for s1, s0 in zip(stamps[1:], stamps[:-1])))
210 freq = 1.0 / mymedian if mymedian else None
211 topicdict[t] = self.TopicTuple(n, c, len(self._qoses.get((t, n)) or []), freq)
212 result = self.TypesAndTopicsTuple(msgtypes, topicdict)
213 if not topics or set(topics) == set(t for t, _, _ in self._topics): self._ttinfo = result
214 return result

217 def read_messages(self, topics=None, start_time=None, end_time=None, raw=False):
218 """
219 Yields messages from the bag, optionally filtered by topic and timestamp.
220
221 @param topics list of topics or a single topic to filter by, if at all
222 @param start_time earliest timestamp of message to return, as ROS time or convertible
223 (int/float/duration/datetime/decimal)
224 @param end_time latest timestamp of message to return, as ROS time or convertible
225 (int/float/duration/datetime/decimal)
226 @param raw if true, then returned messages are tuples of
227 (typename, bytes, typehash, typeclass)
228 @return BagMessage namedtuples of
229 (topic, message, timestamp as rospy/rclpy.Time)
230 """
231 if self.closedclosedclosedclosed: raise ValueError("I/O operation on closed file.")
232 if "w" == self._mode: raise io.UnsupportedOperation("read")
233
234 topics = topics if isinstance(topics, list) else [topics] if topics else None
235 start_ns, end_ns = (api.to_nsec(api.to_time(x)) for x in (start_time, end_time))
236 for schema, channel, message in self._reader.iter_messages(topics, start_ns, end_ns):
237 if raw:
238 typekey = (typename, typehash) = self._schematypes[schema.id]
239 if typekey not in self._types:
240 self._types[typekey] = self._make_message_class(schema, message)
241 msg = (typename, message.data, typehash, self._types[typekey])
242 else: msg = self._decode_message(message, channel, schema)
243 api.TypeMeta.make(msg, channel.topic, self)
244 yield self.BagMessage(channel.topic, msg, api.make_time(nsecs=message.publish_time))
245 if self.closedclosedclosedclosed: break # for schema

248 def write(self, topic, msg, t=None, raw=False, **__):
249 """
250 Writes out message to MCAP file.
251
252 @param topic name of topic
253 @param msg ROS1 message
254 @param t message timestamp if not using current ROS time,
255 as ROS time type or convertible (int/float/duration/datetime/decimal)
256 @param raw if true, `msg` is in raw format, (typename, bytes, typehash, typeclass)
257 """
258 if self.closedclosedclosedclosed: raise ValueError("I/O operation on closed file.")
259 if "r" == self._mode: raise io.UnsupportedOperation("write")
260
261 if raw:
262 typename, binary, typehash = msg[:3]
263 cls = msg[-1]
264 typedef = self._typedefs.get((typename, typehash)) or api.get_message_definition(cls)
265 msg = api.deserialize_message(binary, cls)
266 else:
267 with api.TypeMeta.make(msg, topic) as meta:
268 typename, typehash, typedef = meta.typename, meta.typehash, meta.definition
269 topickey, typekey = (topic, typename, typehash), (typename, typehash)
270
271 nanosec = (time.time_ns() if hasattr(time, "time_ns") else int(time.time() * 10**9)) \
272 if t is None else api.to_nsec(api.to_time(t))
273 if api.ROS2:
274 if typekey not in self._schemas:
275 fullname = api.make_full_typename(typename)
276 schema = self._writer.register_msgdef(fullname, typedef)
277 self._schemas[typekey] = schema
278 schema, data = self._schemas[typekey], api.message_to_dict(msg)
279 self._writer.write_message(topic, schema, data, publish_time=nanosec)
280 else:
281 self._writer.write_message(topic, msg, publish_time=nanosec)
282
283 sec = nanosec / 1E9
284 self._start_time = sec if self._start_time is None else min(sec, self._start_time)
285 self._end_time = sec if self._end_time is None else min(sec, self._end_time)
286 self._topics[topickey] = self._topics.get(topickey, 0) + 1
287 self._types.setdefault(typekey, type(msg))
288 self._typedefs.setdefault(typekey, typedef)
289 self._ttinfo = None

360 def _decode_message(self, message, channel, schema):
361 """
362 Returns ROS message deserialized from binary data.
363
364 @param message mcap.records.Message instance
365 @param channel mcap.records.Channel instance for message
366 @param schema mcap.records.Schema instance for message type
367 """
368 cls = self._make_message_class(schema, message, generate=False)
369 if api.ROS2 and not issubclass(cls, types.SimpleNamespace):
370 msg = api.deserialize_message(message.data, cls)
371 else:
372 msg = self._decoder.decode(schema=schema, message=message)
373 if api.ROS2: # MCAP ROS2 message classes need monkey-patching with expected API
374 msg = self._patch_message(msg, *self._schematypes[schema.id])
375 # Register serialized binary, as MCAP does not support serializing its own creations
376 api.TypeMeta.make(msg, channel.topic, self, data=message.data)
377 typekey = self._schematypes[schema.id]
378 if typekey not in self._types: self._types[typekey] = type(msg)
379 return msg

382 def _make_message_class(self, schema, message, generate=True):
383 """
384 Returns message type class, generating if not already available.
385
386 @param schema mcap.records.Schema instance for message type
387 @param message mcap.records.Message instance
388 @param generate generate message class dynamically if not available
389 """
390 typekey = (typename, typehash) = self._schematypes[schema.id]
391 if api.ROS2 and typekey not in self._types:
392 try: # Try loading class from disk for full compatibility
393 cls = api.get_message_class(typename)
394 clshash = api.get_message_type_hash(cls)
395 if typehash == clshash: self._types[typekey] = cls
396 except Exception: pass # ModuleNotFoundError, AttributeError etc
397 if typekey not in self._types and generate:
398 if api.ROS2: # MCAP ROS2 message classes need monkey-patching with expected API
399 msg = self._decoder.decode(schema=schema, message=message)
400 self._types[typekey] = self._patch_message_class(type(msg), typename, typehash)
401 else:
402 typeclses = genpy.dynamic.generate_dynamic(typename, schema.data.decode())
403 self._types[typekey] = typeclses[typename]
404 return self._types.get(typekey)

407 def _patch_message_class(self, cls, typename, typehash):
408 """
409 Patches MCAP ROS2 message class with expected attributes and methods, recursively.
410
411 @param cls ROS message class as returned from mcap_ros2.decoder
412 @param typename ROS message type name, like "std_msgs/Bool"
413 @param typehash ROS message type hash
414 @return patched class
415 """
416 typekey = (typename, typehash)
417 if typekey not in self._typefields:
418 fields = api.parse_definition_fields(typename, self._typedefs[typekey])
419 self._typefields[typekey] = fields
420 self._field_subtypes[typekey] = {n: (s, h) for n, t in fields.items()
421 for s in [api.scalar(t)]
422 if s not in api.ROS_BUILTIN_TYPES
423 for h in [self._type_subtypes[typekey][s]]}
424
425 # mcap_ros2 creates a dynamic class for each message, having __slots__
426 # but no other ROS2 API hooks; even the class module is "mcap_ros2.dynamic".
427 cls.__module__ = typename.split("/", maxsplit=1)[0]
428 cls.SLOT_TYPES = () # rosidl_runtime_py.utilities.is_message() checks for presence
429 cls._fields_and_field_types = dict(self._typefields[typekey])
430 cls.get_fields_and_field_types = message_get_fields_and_field_types
431 cls.__copy____copy__ = copy_with_slots # MCAP message classes lack support for copy
432 cls.__deepcopy____deepcopy__ = deepcopy_with_slots
433
434 return cls

437 def _patch_message(self, message, typename, typehash):
438 """
439 Patches MCAP ROS2 message with expected attributes and methods, recursively.
440
441 @param message ROS message instance as returned from mcap_ros2.decoder
442 @param typename ROS message type name, like "std_msgs/Bool"
443 @param typehash ROS message type hash
444 @return original message patched, or new instance if ROS2 temporal type
445 """
446 result = message
447 # [(field value, (field type name, field type hash), parent, (field name, ?array index))]
448 stack = [(message, (typename, typehash), None, ())]
449 while stack:
450 msg, typekey, parent, path = stack.pop(0)
451 mycls, typename = type(msg), typekey[0]
452
453 if typename in self._temporal_ctors:
454 # Convert temporal types to ROS2 temporal types for grepros to recognize
455 msg2 = self._temporal_ctors[typename](sec=msg.sec, nanosec=msg.nanosec)
456 if msg is message: result = msg2 # Replace input message
457 elif len(path) == 1: setattr(parent, path[0], msg2) # Set scalar field
458 else: getattr(parent, path[0])[path[1]] = msg2 # Set array field element
459 continue # while stack
460
461 self._patch_message_class(mycls, *typekey)
462
463 for name, subtypekey in self._field_subtypes[typekey].items():
464 v = getattr(msg, name)
465 if isinstance(v, list): # Queue each array element for patching
466 stack.extend((x, subtypekey, msg, (name, i)) for i, x in enumerate(v))
467 else: # Queue scalar field for patching
468 stack.append((v, subtypekey, msg, (name, )))
469
470 return result

473 def _populate_meta(self):
474 """Populates bag metainfo."""
475 summary = self._reader.get_summary()
476 self._start_time = summary.statistics.message_start_time / 1E9
477 self._end_time = summary.statistics.message_end_time / 1E9
478
479 defhashes = {} # Cached {type definition full text: type hash}
480 for cid, channel in summary.channels.items():
481 schema = summary.schemas[channel.schema_id]
482 topic, typename = channel.topic, api.canonical(schema.name)
483
484 typedef = schema.data.decode("utf-8") # Full definition including subtype definitions
485 subtypedefs, nesting = api.parse_definition_subtypes(typedef, nesting=True)
486 typehash = channel.metadata.get("md5sum") or \
487 api.calculate_definition_hash(typename, typedef,
488 tuple(subtypedefs.items()))
489 topickey, typekey = (topic, typename, typehash), (typename, typehash)
490
491 qoses = None
492 if channel.metadata.get("offered_qos_profiles"):
493 try: qoses = yaml.safe_load(channel.metadata["offered_qos_profiles"])
494 except Exception as e:
495 ConsolePrinter.warn("Error parsing topic QoS profiles from %r: %s.",
496 channel.metadata["offered_qos_profiles"], e)
497
498 self._topics.setdefault(topickey, 0)
499 self._topics[topickey] += summary.statistics.channel_message_counts[cid]
500 self._typedefs[typekey] = typedef
501 defhashes[typedef] = typehash
502 for t, d in subtypedefs.items(): # Populate subtype definitions and hashes
503 if d in defhashes: h = defhashes[d]
504 else: h = api.calculate_definition_hash(t, d, tuple(subtypedefs.items()))
505 self._typedefs.setdefault((t, h), d)
506 self._type_subtypes.setdefault(typekey, {})[t] = h
507 defhashes[d] = h
508 for t, subtypes in nesting.items(): # Populate all nested type references
509 h = self._type_subtypes[typekey][t]
510 for t2 in subtypes:
511 h2 = self._type_subtypes[typekey][t2]
512 self._type_subtypes.setdefault((t, h), {})[t2] = h2
513
514 if qoses: self._qoses[topickey] = qoses
515 self._schemas[typekey] = schema
516 self._schematypes[schema.id] = typekey

601 and overwrite is valid and file is writable.

604 ok, mcap_ok, mcap_ros_ok = True, bool(mcap), bool(mcap_ros)

621 def emit(self, topic, msg, stamp=None, match=None, index=None):
622 """Writes out message to MCAP file."""
623 if not self.validatevalidate(): raise Exception("invalid")
624 self._ensure_open()
625 stamp, index = self._ensure_stamp_index(topic, msg, stamp, index)
626 kwargs = dict(publish_time=api.to_nsec(stamp), sequence=index)
627 if api.ROS2:
628 with api.TypeMeta.make(msg, topic) as m:
629 typekey = m.typekey
630 if typekey not in self._schemas:
631 fullname = api.make_full_typename(m.typename)
632 self._schemas[typekey] = self._writer.register_msgdef(fullname, m.definition)
633 schema, data = self._schemas[typekey], api.message_to_dict(msg)
634 self._writer.write_message(topic, schema, data, **kwargs)
635 else:
636 self._writer.write_message(topic, msg, **kwargs)
637 super(McapSink, self).emit(topic, msg, stamp, match, index)

640 def close(self):
641 """Closes output file if open."""
642 try:
643 if self._writer:
644 self._writer.finish()
645 self._file.close()
646 self._file, self._writer = None, None
647 finally:
648 if not self._close_printed and self._counts_counts:
649 self._close_printed = True
650 try: sz = common.format_bytes(os.path.getsize(self._filename))
651 except Exception as e:
652 ConsolePrinter.warn("Error getting size of %s: %s", self._filename, e)
653 sz = "error getting size"
654 ConsolePrinter.debug("Wrote %s in %s to %s (%s).",
655 common.plural("message", sum(self._counts_counts.values())),
656 common.plural("topic", self._counts_counts), self._filename, sz)
657 super(McapSink, self).close()

660 def _ensure_open(self):
661 """Opens output file if not already open."""
662 if self._file: return
663
664 filename = self.args.WRITE
665 if not self._overwrite and os.path.isfile(filename) and os.path.getsize(filename):
666 filename = common.unique_path(filename)
667 if self.args.VERBOSE:
668 ConsolePrinter.debug("Making unique filename %r, as %s does not support "
669 "appending.", filename, type(self).__name___)
670 self._filename = filename
671 common.makedirs(os.path.dirname(self._filename))
672 if self.args.VERBOSE:
673 sz = os.path.exists(self._filename) and os.path.getsize(self._filename)
674 action = "Overwriting" if sz and self._overwrite else "Creating"
675 ConsolePrinter.debug("%s %s.", action, self._filename)
676 self._file = open(self._filename, "wb")
677 self._writer = mcap_ros.writer.Writer(self._file)
