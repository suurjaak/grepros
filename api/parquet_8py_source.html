152  def _process_type(self, topic, msg):
153  """Prepares Parquet schema and writer if not existing."""
154  with rosapi.TypeMeta.make(msg, topic) as m:
155  typename, typehash, typekey = (m.typename, m.typehash, m.typekey)
156  if (topic, typename, typehash) not in self._counts and self.args.VERBOSE:
157  ConsolePrinter.debug("Adding topic %s in Parquet output.", topic)
158  if typekey in self._writers: return
159 
160  basedir, basename = os.path.split(self._filebase)
161  pathname = os.path.join(basedir, re.sub(r"\W", "__", "%s__%s" % (typename, typehash)))
162  filename = os.path.join(pathname, basename)
163  if not self._overwrite:
164  filename = unique_path(filename)
165 
166  cols = []
167  scalars = set(x for x in self.COMMON_TYPES if "[" not in x)
168  for path, value, subtype in rosapi.iter_message_fields(msg, scalars=scalars):
169  coltype = self._make_column_type(subtype)
170  cols += [(".".join(path), coltype)]
171  cols += [(c, self._make_column_type(t, fallback="int64" if "time" == t else None))
172  for c, t in self.MESSAGE_TYPE_BASECOLS]
173 
174  if self.args.VERBOSE:
175  sz = os.path.isfile(filename) and os.path.getsize(filename)
176  action = "Overwriting" if sz and self._overwrite else "Adding"
177  ConsolePrinter.debug("%s type %s in Parquet output.", action, typename)
178  makedirs(pathname)
179 
180  schema = pyarrow.schema(cols)
181  writer = pyarrow.parquet.ParquetWriter(filename, schema, **self.WRITER_ARGS)
182  self._caches[typekey] = []
183  self._filenames[typekey] = filename
184  self._schemas[typekey] = schema
185  self._writers[typekey] = writer

188  def _process_message(self, topic, stamp, msg):
189  """
190  Converts message to pandas dataframe, adds to cache.
191 
192  Writes cache to disk if length reached chunk size.
193  """
194  data = {}
195  typekey = rosapi.TypeMeta.make(msg, topic).typekey
196  for p, v, t in rosapi.iter_message_fields(msg, scalars=set(self.COMMON_TYPES)):
197  data[".".join(p)] = self._make_column_value(v, t)
198  data.update(_topic=topic, _timestamp=self._make_column_value(stamp, "time"))
199  data.update(self._extra_basecols)
200  self._caches[typekey].append(data)
201  if len(self._caches[typekey]) >= self.CHUNK_SIZE:

205  def _make_column_type(self, typename, fallback=None):
206  """
207  Returns pyarrow type for ROS type.
208 
209  @param fallback fallback typename to use for lookup if typename not found
210  """
211  scalartype = rosapi.scalar(typename)
212  timetype = rosapi.get_ros_time_category(scalartype)
213  coltype = self.COMMON_TYPES.get(typename)
214 
215  if not coltype and "[" not in typename and scalartype in self.COMMON_TYPES:
216  coltype = self.COMMON_TYPES[scalartype] # Bounded type like "string<=10"
217  if not coltype and scalartype in self.COMMON_TYPES:
218  coltype = pyarrow.list_(self.COMMON_TYPES[scalartype])
219  if not coltype and timetype in self.COMMON_TYPES:
220  if typename != scalartype:
221  coltype = pyarrow.list_(self.COMMON_TYPES[timetype])
222  else:
223  coltype = self.COMMON_TYPES[timetype]

231  def _make_column_value(self, value, typename=None):
232  """Returns column value suitable for adding to Parquet file."""
233  v = value
234  if isinstance(v, (list, tuple)):
235  if v and rosapi.is_ros_time(v[0]):
236  v = [rosapi.to_nsec(x) for x in v]
237  elif rosapi.scalar(typename) not in rosapi.ROS_BUILTIN_TYPES:
238  v = str([rosapi.message_to_dict(m) for m in v])
239  elif pyarrow.binary() == self.COMMON_TYPES.get(typename):
240  v = bytes(bytearray(v)) # Py2/Py3 compatible
241  else:
242  v = list(v) # Ensure lists not tuples
243  elif rosapi.is_ros_time(v):
244  v = rosapi.to_nsec(v)
245  elif typename and typename not in rosapi.ROS_BUILTIN_TYPES:
246  v = str(rosapi.message_to_dict(v))
247  return v

259  def _configure(self):
260  """Parses args.WRITE_OPTIONS."""
261  ok = True
262 
263  # Populate ROS type aliases like "byte" and "char"
264  for rostype in list(self.COMMON_TYPES):
265  alias = rosapi.get_type_alias(rostype)
266  if alias:
267  self.COMMON_TYPES[alias] = self.COMMON_TYPES[rostype]
268  if alias and rostype + "[]" in self.COMMON_TYPES:
269  self.COMMON_TYPES[alias + "[]"] = self.COMMON_TYPES[rostype + "[]"]
270 
271  for k, v in self.args.WRITE_OPTIONS.items():
272  if k.startswith("column-"):
273  # Parse "column-name=rostype:value"
274  try:
275  name, (rostype, value), myok = k[len("column-"):], v.split(":", 1), True
276  if "string" not in rostype:
277  value = json.loads(value)
278  if not name:
279  ok = myok = False
280  ConsolePrinter.error("Invalid name option in %s=%s", k, v)
281  if rostype not in rosapi.ROS_BUILTIN_TYPES:
282  ok = myok = False
283  ConsolePrinter.error("Invalid type option in %s=%s", k, v)
284  if myok:
285  self.MESSAGE_TYPE_BASECOLS.append((name, rostype))
286  self._extra_basecols.append((name, value))
287  except Exception as e:
288  ok = False
289  ConsolePrinter.error("Invalid column option in %s=%s: %s", k, v, e)
290  elif k.startswith("type-"):
291  # Eval pyarrow datatype from value like "float64()" or "list(uint8())"
292  if not k[len("type-"):]:
293  ok = False
294  ConsolePrinter.error("Invalid type option in %s=%s: %s", k, v)
295  continue # for k, v
296  try:
297  arrowtype = eval(compile(v, "", "eval"), {"__builtins__": self.ARROW_TYPES})
298  self.COMMON_TYPES[k[len("type-"):]] = arrowtype
299  except Exception as e:
300  ok = False
301  ConsolePrinter.error("Invalid type option in %s=%s: %s", k, v, e)
302  elif k.startswith("writer-"):
303  if not k[len("writer-"):]:
304  ok = False
305  ConsolePrinter.error("Invalid name in %s=%s: %s", k, v)
306  continue # for k, v
307  try: v = json.loads(v)
308  except Exception: pass
309  self.WRITER_ARGS[k[len("writer-"):]] = v
310  self._configure_ok = ok
