166  def _process_type(self, topic, msg, rootmsg=None):
167  """Prepares Parquet schema and writer if not existing."""
168  rootmsg = rootmsg or msg
169  with rosapi.TypeMeta.make(msg, root=rootmsg) as m:
170  typename, typehash, typekey = (m.typename, m.typehash, m.typekey)
171  if topic and (topic, typename, typehash) not in self._counts and self.args.VERBOSE:
172  ConsolePrinter.debug("Adding topic %s in Parquet output.", topic)
173  if typekey in self._writers: return
174 
175  basedir, basename = os.path.split(self._filebase)
176  pathname = os.path.join(basedir, re.sub(r"\W", "__", "%s__%s" % (typename, typehash)))
177  filename = os.path.join(pathname, basename)
178  if not self._overwrite:
179  filename = common.unique_path(filename)
180 
181  cols = []
182  scalars = set(x for x in self.COMMON_TYPES if "[" not in x)
183  for path, value, subtype in rosapi.iter_message_fields(msg, scalars=scalars):
184  coltype = self._make_column_type(subtype)
185  cols += [(".".join(path), coltype)]
186  MSGCOLS = self.MESSAGE_TYPE_BASECOLS + (self.MESSAGE_TYPE_NESTCOLS if self._nesting else [])
187  cols += [(c, self._make_column_type(t, fallback="int64" if "time" == t else None))
188  for c, t in MSGCOLS + self._extra_basecols]
189 
190  if self.args.VERBOSE:
191  sz = os.path.isfile(filename) and os.path.getsize(filename)
192  action = "Overwriting" if sz and self._overwrite else "Adding"
193  ConsolePrinter.debug("%s type %s in Parquet output.", action, typename)
194  common.makedirs(pathname)
195 
196  schema = pyarrow.schema(cols)
197  writer = pyarrow.parquet.ParquetWriter(filename, schema, **self.WRITER_ARGS)
198  self._caches[typekey] = []
199  self._filenames[typekey] = filename
200  self._schemas[typekey] = schema
201  self._writers[typekey] = writer
202 
203  nesteds = rosapi.iter_message_fields(msg, messages_only=True) if self._nesting else ()
204  for path, submsgs, subtype in nesteds:
205  scalartype = rosapi.scalar(subtype)
206  if subtype == scalartype and "all" != self._nesting:
207  continue # for path
208  subtypehash = not submsgs and self.source.get_message_type_hash(scalartype)
209  if not isinstance(submsgs, (list, tuple)): submsgs = [submsgs]
210  [submsg] = submsgs[:1] or [self.source.get_message_class(scalartype, subtypehash)()]
211  self._process_type(None, submsg, rootmsg)

214  def _process_message(self, topic, index, stamp, msg, match=None,
215  rootmsg=None, parent_type=None, parent_id=None):
216  """
217  Converts message to pandas dataframe, adds to cache.
218 
219  Writes cache to disk if length reached chunk size.
220 
221  If nesting is enabled, processes nested messages for subtypes in message.
222  If IDs are used, returns generated ID.
223  """
224  data, myid, rootmsg = {}, None, (rootmsg or None)
225  if self._idgenerator: myid = next(self._idgenerator)
226  with rosapi.TypeMeta.make(msg, topic, root=rootmsg) as m:
227  typename, typekey = m.typename, m.typekey
228  for p, v, t in rosapi.iter_message_fields(msg, scalars=set(self.COMMON_TYPES)):
229  data[".".join(p)] = self._make_column_value(v, t)
230  data.update(_topic=topic, _timestamp=self._make_column_value(stamp, "time"))
231  if self._idgenerator: data.update(_id=myid)
232  if self._nesting:
233  COLS = [k for k, _ in self.MESSAGE_TYPE_NESTCOLS if "parent" in k]
234  data.update(zip(COLS, [parent_type, parent_id]))
235  data.update(self._extra_basevals)
236  self._caches[typekey].append(data)
237  super(ParquetSink, self).emit(topic, index, stamp, msg, match)
238 
239  subids = {} # {message field path: [ids]}
240  nesteds = rosapi.iter_message_fields(msg, messages_only=True) if self._nesting else ()
241  for path, submsgs, subtype in nesteds:
242  scalartype = rosapi.scalar(subtype)
243  if subtype == scalartype and "all" != self._nesting:
244  continue # for path
245  if isinstance(submsgs, (list, tuple)):
246  subids[path] = []
247  for submsg in submsgs if isinstance(submsgs, (list, tuple)) else [submsgs]:
248  subid = self._process_message(topic, index, stamp, submsg,
249  rootmsg=rootmsg, parent_type=typename, parent_id=myid)
250  if isinstance(submsgs, (list, tuple)):
251  subids[path].append(subid)
252  data.update(subids)
253 
254  if len(self._caches[typekey]) >= self.CHUNK_SIZE:
255  self._write_table(typekey)
256  return myid

269

272  if not coltype and timetype in self.COMMON_TYPES:
273  if typename != scalartype:

275  else:

284  def _make_column_value(self, value, typename=None):
285  """Returns column value suitable for adding to Parquet file."""
286  v = value
287  if isinstance(v, (list, tuple)):
288  noboundtype = rosapi.canonical(typename, unbounded=True)
289  if v and rosapi.is_ros_time(v[0]):
290  v = [rosapi.to_nsec(x) for x in v]
291  elif rosapi.scalar(typename) not in rosapi.ROS_BUILTIN_TYPES:
292  v = str([rosapi.message_to_dict(m) for m in v])
293  elif pyarrow.binary() in (self.COMMON_TYPES.get(typename),
294  self.COMMON_TYPES.get(noboundtype)):
295  v = bytes(bytearray(v)) # Py2/Py3 compatible
296  else:
297  v = list(v) # Ensure lists not tuples
298  elif rosapi.is_ros_time(v):
299  v = rosapi.to_nsec(v)
300  elif typename and typename not in rosapi.ROS_BUILTIN_TYPES:
301  v = str(rosapi.message_to_dict(v))
302  return v

314  def _configure(self):

322  self.COMMON_TYPES[alias] = self.COMMON_TYPES[rostype]

324  self.COMMON_TYPES[alias + "[]"] = self.COMMON_TYPES[rostype + "[]"]

326  for k, v in self.args.WRITE_OPTIONS.items():
327  if k.startswith("column-"):

329  try:
330  name, (rostype, value), myok = k[len("column-"):], v.split(":", 1), True
331  if "string" not in rostype:
332  value = json.loads(value)
333  if not name:
334  ok = myok = False
335  ConsolePrinter.error("Invalid name option in %s=%s", k, v)
336  if rostype not in rosapi.ROS_BUILTIN_TYPES:
337  ok = myok = False
338  ConsolePrinter.error("Invalid type option in %s=%s", k, v)
339  if myok:
340  self._extra_basecols.append((name, rostype))
341  self._extra_basevals.append((name, value))
342  except Exception as e:
343  ok = False
344  ConsolePrinter.error("Invalid column option in %s=%s: %s", k, v, e)
345  elif k.startswith("type-"):
346  # Eval pyarrow datatype from value like "float64()" or "list(uint8())"
347  if not k[len("type-"):]:
348  ok = False
349  ConsolePrinter.error("Invalid type option in %s=%s: %s", k, v)
350  continue # for k, v
351  try:
352  arrowtype = eval(compile(v, "", "eval"), {"__builtins__": self.ARROW_TYPES})
353  self.COMMON_TYPES[k[len("type-"):]] = arrowtype
354  except Exception as e:
355  ok = False
356  ConsolePrinter.error("Invalid type option in %s=%s: %s", k, v, e)
357  elif k.startswith("writer-"):
358  if not k[len("writer-"):]:
359  ok = False
360  ConsolePrinter.error("Invalid name in %s=%s: %s", k, v)
361  continue # for k, v
362  try: v = json.loads(v)
363  except Exception: pass
364  self.WRITER_ARGS[k[len("writer-"):]] = v
365  return ok

368  def _configure_ids(self):
369  """Configures ID generator from args.WRITE_OPTIONS, returns success."""
370  ok = True
371 
372  try: integer_types, text_types = (int, long), (str, unicode) # Py2
373  except Exception: integer_types, text_types = (int, ), (bytes, str) # Py3
374 
375  k, v = "idgenerator", self.args.WRITE_OPTIONS.get("idgenerator")
376  if k in self.args.WRITE_OPTIONS:
377  val, ex, ns = v, None, dict(self.ARROW_TYPES, itertools=itertools, uuid=uuid)
378  for root in v.split(".", 1)[:1]:
379  try: ns[root] = common.import_item(root) # Provide root module
380  except Exception: pass
381  try: common.import_item(re.sub(r"\(.+", "", v)) # Ensure nested imports
382  except Exception: pass
383  try: val = eval(compile(v, "", "eval"), ns)
384  except Exception as e: ok, ex = False, e
385  if isinstance(val, text_types): ok = False
386 
387  if ok:
388  try: self._idgenerator = iter(val)
389  except Exception as e:
390  try: self._idgenerator = iter(val, self) # (callable=val, sentinel=self)
391  except Exception as e: ok, ex = False, e
392  if not ok:
393  ConsolePrinter.error("Invalid value in %s=%s%s", k, v, (": %s" % ex if ex else ""))
394  elif not self._nesting:
395  self.MESSAGE_TYPE_BASECOLS.append(("_id", "string"))
396 
397  if ok and self._idgenerator: # Detect given ID column type
398  fval, typename, generator = next(self._idgenerator), None, self._idgenerator
399  if rosapi.is_ros_time(fval):
400  typename = "time" if "time" in str(type(fval)).lower() else "duration"
401  elif isinstance(fval, integer_types):
402  typename = "int64"
403  elif isinstance(fval, float):
404  typename = "float64"
405  elif not isinstance(fval, str): # Cast whatever it is to string
406  fval, self._idgenerator = str(fval), (str(x) for x in generator)
407  if typename:
408  repl = lambda n, t: (n, typename) if "_id" in n else (n, t)
409  self.MESSAGE_TYPE_BASECOLS = [repl(*x) for x in self.MESSAGE_TYPE_BASECOLS]
410  self.MESSAGE_TYPE_NESTCOLS = [repl(*x) for x in self.MESSAGE_TYPE_NESTCOLS]
411  self._idgenerator = itertools.chain([fval], self._idgenerator)
412  return ok
