214  def __new__(cls, filename, mode="r", decompress=False, reindex=False, progress=False):
215  """
216  Returns an object for reading or writing ROS bags.
217 
218  Result is rosbag.Bag in ROS1, or an object with a partially conforming API
219  if using embag in ROS1, or if using ROS2.
220 
221  Plugins can add their own format support to READER_CLASSES and WRITER_CLASSES.
222  Classes can have a static/class method `autodetect(filename)`
223  returning whether given file is readable for the plugin class.
224 
225  Extra methods compared with rosbag.Bag: get_message_class(),
226  get_message_definition(), get_message_type_hash(), and get_topic_info().
227 
228  @param mode return reader if "r" else writer
229  @param decompress decompress archived bag to file directory
230  @param reindex reindex unindexed bag (ROS1 only), making a backup if indexed format
231  @param progress show progress bar with decompression or reindexing status
232  """
233  if Decompressor.is_compressed(filename):
234  if decompress: filename = Decompressor.decompress(filename, progress)
235  else: raise Exception("decompression not enabled")
236 
237  if "a" == mode and (not os.path.exists(filename) or not os.path.getsize(filename)):
238  mode = "w" # rosbag raises error on append if no file or empty file
239  os.path.exists(filename) and os.remove(filename)
240  classes = set(cls.READER_CLASSES if "r" == mode else cls.WRITER_CLASSES)
241  for detect, mycls in ((d, c) for d in (True, False) for c in list(classes)):
242  use, discard = not detect, False
243  try: # Try auto-detecting suitable class first
244  if detect and callable(getattr(mycls, "autodetect", None)):
245  use, discard = mycls.autodetect(filename), True
246  if use:
247  result = mycls(filename, mode=mode, reindex=reindex, progress=progress)
248  if result: return result
249  except Exception as e:
250  discard = True
251  ConsolePrinter.warn("Failed to open %r for %s with %s: %s.",
252  filename, "reading" if "r" == mode else "writing", mycls, e)
253  discard and classes.discard(mycls)
254  raise Exception("No suitable %s class available" % ("reader" if "r" == mode else "writer"))

306 @memoize
307 def calculate_definition_hash(typename, msgdef, extradefs=()):
308  """
309  Returns MD5 hash for message type definition.
310 
311  @param extradefs additional subtype definitions as ((typename, msgdef), )
312  """
313  # "type name (= constvalue)?" or "type name (defaultvalue)?" (ROS2 format)
314  FIELD_RGX = re.compile(r"^([a-z][^\s:]+)\s+([^\s=]+)(\s*=\s*([^\n]+))?(\s+([^\n]+))?", re.I)
315  STR_CONST_RGX = re.compile(r"^w?string\s+([^\s=#]+)\s*=")
316  lines, pkg = [], typename.rsplit("/", 1)[0]
317  subtypedefs = dict(extradefs, **parse_definition_subtypes(msgdef))
318  extradefs = tuple(subtypedefs.items())
319 
320  # First pass: write constants
321  for line in msgdef.splitlines():
322  if set(line) == set("="): # Subtype separator
323  break # for line
324  # String constants cannot have line comments
325  if "#" in line and not STR_CONST_RGX.match(line): line = line[:line.index("#")]
326  match = FIELD_RGX.match(line)
327  if match and match.group(3):
328  lines.append("%s %s=%s" % (match.group(1), match.group(2), match.group(4).strip()))
329  # Second pass: write fields and subtype hashes
330  for line in msgdef.splitlines():
331  if set(line) == set("="): # Subtype separator
332  break # for line
333  if "#" in line and not STR_CONST_RGX.match(line): line = line[:line.index("#")]
334  match = FIELD_RGX.match(line)
335  if match and not match.group(3): # Not constant
336  scalartype, namestr = scalar(match.group(1)), match.group(2)
337  if scalartype in ROS_COMMON_TYPES:
338  typestr = match.group(1)
339  if match.group(5): namestr = (namestr + " " + match.group(6)).strip()
340  else:
341  subtype = scalartype if "/" in scalartype else "std_msgs/Header" \
342  if "Header" == scalartype else "%s/%s" % (pkg, scalartype)
343  typestr = calculate_definition_hash(subtype, subtypedefs[subtype], extradefs)
344  lines.append("%s %s" % (typestr, namestr))
345  return hashlib.md5("\n".join(lines).encode()).hexdigest()

473 def iter_message_fields(msg, messages_only=False, scalars=(), top=()):
474  """
475  Yields ((nested, path), value, typename) from ROS message.
476 
477  @param messages_only whether to yield only values that are ROS messages themselves
478  or lists of ROS messages, else will yield scalar and list values
479  @param scalars sequence of ROS types to consider as scalars, like ("time", duration")
480  """
481  fieldmap = realapi.get_message_fields(msg)
482  if fieldmap is msg: return
483  if messages_only:
484  for k, t in fieldmap.items():
485  v, scalart = realapi.get_message_value(msg, k, t), realapi.scalar(t)
486  is_sublist = isinstance(v, (list, tuple)) and scalart not in ROS_COMMON_TYPES
487  is_forced_scalar = get_ros_time_category(scalart) in scalars
488  if not is_forced_scalar and realapi.is_ros_message(v):
489  for p2, v2, t2 in iter_message_fields(v, True, scalars, top=top + (k, )):
490  yield p2, v2, t2
491  if is_forced_scalar or is_sublist or realapi.is_ros_message(v, ignore_time=True):
492  yield top + (k, ), v, t
493  else:
494  for k, t in fieldmap.items():
495  v = realapi.get_message_value(msg, k, t)
496  is_forced_scalar = get_ros_time_category(realapi.scalar(t)) in scalars
497  if not is_forced_scalar and realapi.is_ros_message(v):
498  for p2, v2, t2 in iter_message_fields(v, False, scalars, top=top + (k, )):

542 def make_message_hash(msg, include=(), exclude=()):
543  """
544  Returns hashcode for ROS message, as a hex digest.
545 
546  @param include message fields to include if not all, as [((nested, path), re.Pattern())]
547  @param exclude message fields to exclude, as [((nested, path), re.Pattern())]
548  """
549  hasher = hashlib.md5()
550 
551  def walk_message(obj, top=()):
552  fieldmap = get_message_fields(obj)
553  fieldmap = filter_fields(fieldmap, include=include, exclude=exclude)
554  for k, t in fieldmap.items():
555  v, path = get_message_value(obj, k, t), top + (k, )
556  if is_ros_message(v):
557  walk_message(v, path)
558  elif isinstance(v, (list, tuple)) and scalar(t) not in ROS_BUILTIN_TYPES:
559  for x in v: walk_message(x, path)
560  else:
561  s = "%s=%s" % (path, v)
562  hasher.update(s.encode("utf-8", errors="backslashreplace"))
563  if not hasattr(obj, "__slots__"):
564  s = "%s=%s" % (top, obj)
565  hasher.update(s.encode("utf-8", errors="backslashreplace"))
566 
567  walk_message(msg)
568  return hasher.hexdigest()

600 
601  Dpes not recurse into subtypes.
602

604  @param typedef ROS message definition, like "Header header\nbool a\nMyCls2 b"

607  """
608  result = collections.OrderedDict() # {subtypename: subtypedef}
609 
610  FIELD_RGX = re.compile(r"^([a-z][^\s:]+)\s+([^\s=]+)(\s*=\s*([^\n]+))?(\s+([^\n]+))?", re.I)
611  STR_CONST_RGX = re.compile(r"^w?string\s+([^\s=#]+)\s*=")

613  for line in filter(bool, typedef.splitlines()):
614  if set(line) == set("="): # Subtype separator
615  break # for line
616  if "#" in line and not STR_CONST_RGX.match(line): line = line[:line.index("#")]
617  match = FIELD_RGX.match(line)
618  if not match or match.group(3): # Constant or not field
619  continue # for line
620 
621  name, typename, scalartype = match.group(2), match.group(1), scalar(match.group(1))
622  if scalartype not in ROS_COMMON_TYPES:
623  pkg2 = "" if "/" in scalartype else "std_msgs" if "Header" == scalartype else pkg
624  typename = "%s/%s" % (pkg2, typename) if pkg2 else typename
625  result[name] = typename

636  @param nesting whether to additionally return type nesting information as

638  @return {"pkg/MsgType": "full definition for MsgType including subtypes"}
639  or ({typedefs}, {nesting}) if nesting
640  """
641  result = collections.OrderedDict() # {subtypename: subtypedef}
642  nesteds = collections.defaultdict(list) # {subtypename: [subtypename2, ]})
643 
644  # Parse individual subtype definitions from full definition
645  curtype, curlines = "", []
646  # Separator line, and definition header like 'MSG: std_msgs/MultiArrayLayout'
647  rgx = re.compile(r"^((=+)|(MSG: (.+)))$") # Group 2: separator, 4: new type
648  for line in typedef.splitlines():
649  m = rgx.match(line)
650  if m and m.group(2) and curtype: # Separator line between nested definitions
651  result[curtype] = "\n".join(curlines)
652  curtype, curlines = "", []
653  elif m and m.group(4): # Start of nested definition "MSG: pkg/MsgType"
654  curtype, curlines = m.group(4), []
655  elif not m and curtype: # Definition content
656  curlines.append(line)
657  if curtype:
658  result[curtype] = "\n".join(curlines)
659 
660  # "type name (= constvalue)?" or "type name (defaultvalue)?" (ROS2 format)
661  FIELD_RGX = re.compile(r"^([a-z][^\s]+)\s+([^\s=]+)(\s*=\s*([^\n]+))?(\s+([^\n]+))?", re.I)
662  # Concatenate nested subtype definitions to parent subtype definitions
663  for subtype, subdef in list(result.items()):
664  pkg, seen = subtype.rsplit("/", 1)[0], set()
665  for line in subdef.splitlines():
666  m = FIELD_RGX.match(line)
667  if m and m.group(1):
668  scalartype, fulltype = realapi.scalar(m.group(1)), None
669  if scalartype not in ROS_COMMON_TYPES:
670  fulltype = scalartype if "/" in scalartype else "std_msgs/Header" \
671  if "Header" == scalartype else "%s/%s" % (pkg, scalartype)
672  if fulltype in result and fulltype not in seen:
673  addendum = "%s\nMSG: %s\n%s" % ("=" * 80, fulltype, result[fulltype])
674  result[subtype] = result[subtype].rstrip() + ("\n\n%s\n" % addendum)
675  nesteds[subtype].append(fulltype)
676  seen.add(fulltype)
677  return (result, nesteds) if nesting else result

683 
684  Returns type unchanged if an ordinary type. In ROS2, returns unbounded type,
685  e.g. "string" from "string<=10[<=5]".
686  """
