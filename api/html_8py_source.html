49 def __init__(self, args=None, **kwargs):
50 """
51 @param args arguments as namespace or dictionary, case-insensitive;
52 or a single path as the name of HTML file to write
53 @param args.write name of HTML file to write,
54 will add counter like .2 to filename if exists
55 @param args.write_options ```
56 {"template": path to custom HTML template, if any,
57 "overwrite": whether to overwrite existing file
58 (default false)}
59 ```
60 @param args.highlight highlight matched values (default true)
61 @param args.orderby "topic" or "type" if any to group results by
62 @param args.color False or "never" for not using colors in replacements
63 @param args.emit_field message fields to emit if not all
64 @param args.noemit_field message fields to skip in output
65 @param args.max_field_lines maximum number of lines to output per field
66 @param args.start_line message line number to start output from
67 @param args.end_line message line number to stop output at
68 @param args.max_message_lines maximum number of lines to output per message
69 @param args.lines_around_match number of message lines around matched fields to output
70 @param args.matched_fields_only output only the fields where match was found
71 @param args.wrap_width character width to wrap message YAML output at
72 @param args.match_wrapper string to wrap around matched values,
73 both sides if one value, start and end if more than one,
74 or no wrapping if zero values
75 @param args.meta whether to emit metainfo
76 @param args.verbose whether to emit debug information
77 @param kwargs any and all arguments as keyword overrides,
78 case-insensitive

106 def emit(self, topic, msg, stamp=None, match=None, index=None):
107 """Writes message to output file."""
108 if not self.validatevalidate(): raise Exception("invalid")
109 stamp, index = self._ensure_stamp_index(topic, msg, stamp, index)
110 self._queue.put((topic, msg, stamp, match, index))
111 if not self._writer:
112 self._writer = threading.Thread(target=self._stream)
113 self._writer.start()
114 if self._queue.qsize() > 100: self._queue.join()

116 def validate(self):
117 """
118 Returns whether write options are valid and ROS environment is set and file is writable,
119 emits error if not.
120 """
121 if self.validvalid is not None: return self.validvalid
122 result = True
123 if self.args.WRITE_OPTIONS.get("template") and not os.path.isfile(self._template_path):
124 result = False
125 ConsolePrinter.error("Template does not exist: %s.", self._template_path)
126 if self.args.WRITE_OPTIONS.get("overwrite") not in (None, True, False, "true", "false"):
127 ConsolePrinter.error("Invalid overwrite option for HTML: %r. "
128 "Choose one of {true, false}.",
129 self.args.WRITE_OPTIONS["overwrite"])
130 result = False
131 if not common.verify_io(self.args.WRITE, "w"):
132 result = False
133 self.validvalid = api.validate() and result

136 def close(self):
137 """Closes output file, if any."""
138 try:
139 if self._writer:
140 writer, self._writer = self._writer, None
141 self._queue.put(None)
142 writer.is_alive() and writer.join()
143 finally:
144 if not self._close_printed and self._counts_counts:
145 self._close_printed = True
146 try: sz = common.format_bytes(os.path.getsize(self._filename))
147 except Exception as e:
148 ConsolePrinter.warn("Error getting size of %s: %s", self._filename, e)
149 sz = "error getting size"
150 ConsolePrinter.debug("Wrote %s in %s to %s (%s).",
151 plural("message", sum(self._counts_counts.values())),
152 plural("topic", self._counts_counts), self._filename, sz)
153 super(HtmlSink, self).close()

159 def format_message(self, msg, highlight=False):
160 """Returns message as formatted string, optionally highlighted for matches if configured."""
161 text = TextSinkMixin.format_message(self, msg, self.args.HIGHLIGHT and highlight)

170 def _stream(self):
171 """Writer-loop, streams HTML template to file."""
172 if not self._writer:
173 return
174
175 try:
176 with open(self._template_path, "r") as f: tpl = f.read()
177 template = step.Template(tpl, escape=True, strip=False)
178 ns = dict(source=self.source, sink=self, messages=self._produce(),
179 args=None, timeline=not self.args.ORDERBY)
180 if main.CLI_ARGS: ns.update(args=main.CLI_ARGS)
181 common.makedirs(os.path.dirname(self.args.WRITE))
182 if not self._overwrite:
183 self._filename = common.unique_path(self.args.WRITE, empty_ok=True)
184 if self.args.VERBOSE:
185 sz = os.path.isfile(self._filename) and os.path.getsize(self._filename)
186 action = "Overwriting" if sz and self._overwrite else "Creating"
187 ConsolePrinter.debug("%s %s.", action, self._filename)
188 with open(self._filename, "wb") as f:
189 template.stream(f, ns, unbuffered=True)
190 except Exception as e:
191 self.thread_excepthook("Error writing HTML output %r: %r" % (self._filename, e), e)
192 finally:
193 self._writer = None

195 def _produce(self):
196 """Yields messages from emit queue, as (topic, msg, stamp, match, index)."""
197 while True:
198 entry = self._queue.get()
199 if entry is None:
200 self._queue.task_done()
201 break # while
202 (topic, msg, stamp, match, index) = entry
203 topickey = api.TypeMeta.make(msg, topic).topickey
204 if self.args.VERBOSE and topickey not in self._counts_counts:
205 ConsolePrinter.debug("Adding topic %s in HTML output.", topic)
206 yield entry
207 super(HtmlSink, self).emit(topic, msg, stamp, match, index)
208 self._queue.task_done()
209 try:
210 while self._queue.get_nowait() or True: self._queue.task_done()
211 except queue.Empty: pass
