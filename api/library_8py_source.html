68def grep(args=None, **kwargs):
69 """
70 Yields matching messages from specified source.
71
72 Initializes grepros if not already initialized.
73
74 Read from bagfiles: `grep(file="2022-10-*.bag", pattern="cpu")`.
75
76 Read from live topics: `grep(live=True, pattern="cpu")`.
77
78
79 @param args arguments as namespace or dictionary, case-insensitive;
80 or a single path as the ROS bagfile to read,
81 or one or more {@link grepros.api.Bag Bag} instances
82 @param kwargs any and all arguments as keyword overrides, case-insensitive
83 <!--sep-->
84
85 Bag source:
86 @param args.file names of ROS bagfiles to read if not all in directory
87 @param args.path paths to scan if not current directory
88 @param args.recurse recurse into subdirectories when looking for bagfiles
89 @param args.decompress decompress archived bags to file directory
90 @param args.reindex make a copy of unindexed bags and reindex them (ROS1 only)
91 @param args.orderby "topic" or "type" if any to group results by
92 @param args.bag one or more {@link grepros.api.Bag Bag} instances
93 <!--sep-->
94
95 Live source:
96 @param args.live whether reading messages from live ROS topics
97 @param args.queue_size_in subscriber queue size (default 10)
98 @param args.ros_time_in stamp messages with ROS time instead of wall time
99 <!--sep-->
100
101 App source:
102 @param args.app whether reading messages from iterable or pushed data;
103 may contain the iterable itself
104 @param args.iterable iterable yielding (topic, msg, stamp) or (topic, msg);
105 yielding `None` signals end of content
106 Any source:
107 @param args.topic ROS topics to read if not all
108 @param args.type ROS message types to read if not all
109 @param args.skip_topic ROS topics to skip
110 @param args.skip_type ROS message types to skip
111 @param args.start_time earliest timestamp of messages to read
112 @param args.end_time latest timestamp of messages to read
113 @param args.start_index message index within topic to start from
114 @param args.end_index message index within topic to stop at
115
116 @param args.nth_message read every Nth message in topic
117 @param args.nth_interval minimum time interval between messages in topic
118
119 @param args.select_field message fields to use in matching if not all
120 @param args.noselect_field message fields to skip in matching
121
122 @param args.unique emit messages that are unique in topic
123 (select_field and noselect_field apply if specified)
124 @param args.condition Python expressions that must evaluate as true
125 for message to be processable, see ConditionMixin
126 <!--sep-->
127
128 Search&zwj;:
129 @param args.pattern pattern(s) to find in message field values
130 @param args.fixed_string pattern contains ordinary strings, not regular expressions
131 @param args.case use case-sensitive matching in pattern
132 @param args.invert select messages not matching pattern
133
134 @param args.nth_match emit every Nth match in topic
135 @param args.max_count number of matched messages to emit (per file if bag input)
136 @param args.max_per_topic number of matched messages to emit from each topic
137 @param args.max_topics number of topics to print matches from
138
139 @param args.before number of messages of leading context to emit before match
140 @param args.after number of messages of trailing context to emit after match
141 @param args.context number of messages of leading and trailing context
142 to emit around match
143
144 @param args.highlight highlight matched values
145 @param args.match_wrapper string to wrap around matched values,
146 both sides if one value, start and end if more than one,
147 or no wrapping if zero values
148
149 @return {@link grepros.Scanner.GrepMessage GrepMessage} namedtuples
150 of (topic, message, timestamp, match, index)
151 """
152 DEFAULT_ARGS = dict(FILE=[], LIVE=False, APP=False, ITERABLE=None,
153 COLOR="never", HIGHLIGHT=False)
154
155 args0 = args
156 is_bag = isinstance(args, Bag) or \
157 common.is_iterable(args) and all(isinstance(x, Bag) for x in args)
158 args = {"FILE": str(args)} if isinstance(args, common.PATH_TYPES) else \
159 {} if is_bag or isinstance(args, Source) else args
160 args = common.ensure_namespace(args, DEFAULT_ARGS, **kwargs)
161 main.validate_args(main.process_args(args))
162 if not _inited: init(args)
163
164 if common.is_iterable(args.APP) and not common.is_iterable(args.ITERABLE):
165 args.APP, args.ITERABLE = True, args.APP
166 src = args0 if isinstance(args0, Source) else \
167 TopicSource(args) if args.LIVE else \
168 AppSource(args) if args.APP else \
169 BagSource(args0, **vars(args)) if is_bag else BagSource(args)
170 src.validate()
171
172 try:
173 for x in Scanner(args).find(src): yield x
174 finally:
175 if not isinstance(args0, (Bag, Source)): src.close()

178def source(args=None, **kwargs):
179 """
180 Convenience for creating a {@link grepros.inputs.Source Source} instance from arguments.
181
182 Initializes grepros if not already initialized.
183
184 @param args arguments as namespace or dictionary, case-insensitive;
185 or a single path as the ROS bagfile to read
186 @param kwargs any and all arguments as keyword overrides, case-insensitive
187 @param args.file one or more names of ROS bagfiles to read from
188 @param args.live read messages from live ROS topics instead
189 @param args.app read messages from iterable or pushed data instead;
190 may contain the iterable itself
191 <!--sep-->
192
193 Bag source:
194 @param args.file names of ROS bagfiles to read if not all in directory
195 @param args.path paths to scan if not current directory
196 @param args.recurse recurse into subdirectories when looking for bagfiles
197 @param args.orderby "topic" or "type" if any to group results by
198 @param args.decompress decompress archived bags to file directory
199 @param args.reindex make a copy of unindexed bags and reindex them (ROS1 only)
200 @param args.progress whether to print progress bar
201 <!--sep-->
202
203 Live source:
204 @param args.queue_size_in subscriber queue size (default 10)
205 @param args.ros_time_in stamp messages with ROS time instead of wall time
206 @param args.progress whether to print progress bar
207 <!--sep-->
208
209 App source:
210 @param args.iterable iterable yielding (topic, msg, stamp) or (topic, msg);
211 yielding `None` signals end of content
212 <!--sep-->
213
214 Any source:
215 @param args.topic ROS topics to read if not all
216 @param args.type ROS message types to read if not all
217 @param args.skip_topic ROS topics to skip
218 @param args.skip_type ROS message types to skip
219 @param args.start_time earliest timestamp of messages to read
220 @param args.end_time latest timestamp of messages to read
221 @param args.start_index message index within topic to start from
222 @param args.end_index message index within topic to stop at
223 @param args.unique emit messages that are unique in topic
224 @param args.select_field message fields to use for uniqueness if not all
225 @param args.noselect_field message fields to skip for uniqueness
226 @param args.nth_message read every Nth message in topic
227 @param args.nth_interval minimum time interval between messages in topic
228 @param args.condition Python expressions that must evaluate as true
229 for message to be processable, see ConditionMixin
230 """
231 DEFAULT_ARGS = dict(FILE=[], LIVE=False, APP=False, ITERABLE=None)
232 args = {"FILE": str(args)} if isinstance(args, common.PATH_TYPES) else args
233 args = common.ensure_namespace(args, DEFAULT_ARGS, **kwargs)
234 if not _inited: init(args)
235
236 if common.is_iterable(args.APP) and not common.is_iterable(args.ITERABLE):
237 args.APP, args.ITERABLE = True, args.APP
238 result = (TopicSource if args.LIVE else AppSource if args.APP else BagSource)(args)
239 result.validate()
240 return result

243def sink(args=None, **kwargs):
244 """
245 Convenience for creating a {@link grepros.outputs.Sink Sink} instance from arguments,
246 {@link grepros.outputs.MultiSink MultiSink} if several outputs.
247
248 Initializes grepros if not already initialized.
249
250 @param args arguments as namespace or dictionary, case-insensitive;
251 or a single item as sink target like bag filename
252 @param kwargs any and all arguments as keyword overrides, case-insensitive
253 @param args.app provide messages to given callback function
254 @param args.console print matches to console
255 @param args.publish publish matches to live topics
256 @param args.write file or other target like Postgres database to write,
257 as "target", or ["target", dict(format="format", ..)]
258 or [[..target1..], [..target2..], ..]
259 @param args.write_options format-specific options like
260 {"overwrite": whether to overwrite existing file
261 (default false)}
262 <!--sep-->
263
264 Console sink:
265 @param args.line_prefix print source prefix like bag filename on each message line
266 @param args.max_field_lines maximum number of lines to print per field
267 @param args.start_line message line number to start output from
268 @param args.end_line message line number to stop output at
269 @param args.max_message_lines maximum number of lines to output per message
270 @param args.lines_around_match number of message lines around matched fields to output
271 @param args.matched_fields_only output only the fields where match was found
272 @param args.wrap_width character width to wrap message YAML output at
273 @param args.match_wrapper string to wrap around matched values,
274 both sides if one value, start and end if more than one,
275 or no wrapping if zero values
276 <!--sep-->
277
278 Console / HTML sink:
279 @param args.color False or "never" for not using colors in replacements
280 @param args.highlight highlight matched values (default true)
281 @param args.emit_field message fields to emit if not all
282 @param args.noemit_field message fields to skip in output
283 @param args.max_field_lines maximum number of lines to output per field
284 @param args.start_line message line number to start output from
285 @param args.end_line message line number to stop output at
286 @param args.max_message_lines maximum number of lines to output per message
287 @param args.lines_around_match number of message lines around matched fields to output
288 @param args.matched_fields_only output only the fields where match was found
289 @param args.wrap_width character width to wrap message YAML output at
290 @param args.match_wrapper string to wrap around matched values,
291 both sides if one value, start and end if more than one,
292 or no wrapping if zero values
293 <!--sep-->
294
295 Topic sink:
296 @param args.queue_size_out publisher queue size (default 10)
297 @param args.publish_prefix output topic prefix, prepended to input topic
298 @param args.publish_suffix output topic suffix, appended to output topic
299 @param args.publish_fixname single output topic name to publish to,
300 overrides prefix and suffix if given
301 <!--sep-->
302
303 App sink:
304 @param args.emit callback(topic, msg, stamp, highlighted msg, index in topic)
305 if any
306 @param args.metaemit callback(metadata dict) if any,
307 invoked before first emit from source batch
308 <!--sep-->
309
310 Any sink:
311 @param args.meta whether to print metainfo
312 @param args.verbose whether to print debug information
313 """
314 DEFAULT_ARGS = dict(CONSOLE=False, PUBLISH=False, WRITE=[], APP=False, EMIT=None, METAEMIT=None)
315
316 result = None
317 args = {"WRITE": str(args)} if isinstance(args, common.PATH_TYPES) else args
318 args = common.ensure_namespace(args, DEFAULT_ARGS, **kwargs)
319 if not _inited: init(args)
320
321 if args.WRITE:
322 if isinstance(args.WRITE, common.PATH_TYPES):
323 args.WRITE = [[args.WRITE]] # Nest deeper, single file given
324 elif isinstance(args.WRITE, (list, tuple)) and isinstance(args.WRITE[0], common.PATH_TYPES):
325 args.WRITE = [args.WRITE] # Nest deeper, must have been single [target, ..opts]
326 if callable(args.APP) and not callable(args.EMIT): args.APP, args.EMIT = True, args.APP
327
328 multisink = MultiSink(args)
329 multisink.validate()
330 result = multisink.sinks[0] if len(multisink.sinks) == 1 else multisink
331 return result
