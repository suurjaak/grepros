27class SqliteSink(BaseDataSink):
28 """
29 Writes messages to an SQLite database.
30
31 Output will have:
32 - table "messages", with all messages as serialized binary data
33 - table "types", with message definitions
34 - table "topics", with topic information
35
36 plus:
37 - table "pkg/MsgType" for each message type, with detailed fields,
38 and JSON fields for arrays of nested subtypes,
39 with foreign keys if nesting else subtype values as JSON dictionaries;
40 plus underscore-prefixed fields for metadata, like `_topic` as the topic name.
41
42 If launched with nesting-option, tables will also be created for each
43 nested message type.
44
45 - view "/topic/full/name" for each topic,
46 selecting from the message type table
47
48 """

111 def _init_db(self):
112 """Opens the database file and populates schema if not already existing."""
113 for t in (dict, list, tuple): sqlite3.register_adapter(t, json.dumps)
114 for t in six.integer_types:
115 sqlite3.register_adapter(t, lambda x: str(x) if abs(x) > self.MAX_INT else x)
116 sqlite3.register_converter("JSON", json.loads)
117 if self.args.VERBOSE:
118 sz = os.path.exists(self._filename) and os.path.getsize(self._filename)
119 action = "Overwriting" if sz and self._overwrite else \
120 "Appending to" if sz else "Creating"
121 ConsolePrinter.debug("%s %s%s.", action, self._filename,
122 (" (%s)" % format_bytes(sz)) if sz else "")
123 super(SqliteSink, self)._init_db()

126 def _load_schema(self):
127 """Populates instance attributes with schema metainfo."""
128 super(SqliteSink, self)._load_schema()
129 for row in self.db.execute("SELECT name FROM sqlite_master "
130 "WHERE type = 'table' AND name LIKE '%/%'"):
131 cols = self.db.execute("PRAGMA table_info(%s)" % quote(row["name"])).fetchall()
132 typerow = next((x for x in self._types.values()
133 if x["table_name"] == row["name"]), None)
134 if not typerow: continue # for row
135 typekey = (typerow["type"], typerow["md5"])
136 self._schema[typekey] = collections.OrderedDict([(c["name"], c) for c in cols])

139 def _process_message(self, topic, msg, stamp):
140 """Inserts message to messages-table, and to pkg/MsgType tables."""
141 with api.TypeMeta.make(msg, topic) as m:
142 topic_id, typename = self._topics[m.topickey]["id"], m.typename
143 margs = dict(dt=api.to_datetime(stamp), timestamp=api.to_nsec(stamp),
144 topic=topic, name=topic, topic_id=topic_id, type=typename,
145 yaml=str(msg) if self._do_yaml else "", data=api.serialize_message(msg))
146 self._ensure_execute(self._get_dialect_option("insert_message"), margs)
147 super(SqliteSink, self)._process_message(topic, msg, stamp)

150 def _connect(self):
151 """Returns new database connection."""
152 makedirs(os.path.dirname(self._filename))
153 if self._overwrite: open(self._filename, "w").close()
154 db = sqlite3.connect(self._filename, check_same_thread=False,
155 detect_types=sqlite3.PARSE_DECLTYPES)
156 if not self.COMMIT_INTERVALCOMMIT_INTERVAL: db.isolation_level = None
157 db.row_factory = lambda cursor, row: dict(sqlite3.Row(cursor, row))
158 return db

176 def _get_next_id(self, table):
177 """Returns next ID value for table, using simple auto-increment."""
178 if not self._id_counters.get(table):
179 sql = "SELECT COALESCE(MAX(_id), 0) AS id FROM %s" % quote(table)
180 self._id_counters[table] = self.db.execute(sql).fetchone()["id"]
181 self._id_counters[table] += 1
182 return self._id_counters[table]
