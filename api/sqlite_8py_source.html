26 class SqliteSink(DataSinkBase):
27  """
28  Writes messages to an SQLite database.
29 
30  Output will have:
31  - table "messages", with all messages as serialized binary data
32  - table "types", with message definitions
33  - table "topics", with topic information
34 
35  plus:
36  - table "pkg/MsgType" for each message type, with detailed fields,
37  and JSON fields for arrays of nested subtypes,
38  with foreign keys if nesting else subtype values as JSON dictionaries;
39  plus underscore-prefixed fields for metadata, like `_topic` as the topic name.
40 
41  If launched with nesting-option, tables will also be created for each

43 
44  - view "/topic/full/name" for each topic,

99  def _init_db(self):
100  """Opens the database file and populates schema if not already existing."""
101  for t in (dict, list, tuple): sqlite3.register_adapter(t, json.dumps)
102  sqlite3.register_adapter(int, lambda x: str(x) if abs(x) > self.MAX_INT else x)
103  if sys.version_info < (3, ):
104  sqlite3.register_adapter(long, lambda x: str(x) if abs(x) > self.MAX_INT else x)

115  def _load_schema(self):
116  """Populates instance attributes with schema metainfo."""
117  super(SqliteSink, self)._load_schema()
118  for row in self._db.execute("SELECT name FROM sqlite_master "
119  "WHERE type = 'table' AND name LIKE '%/%'"):
120  cols = self._db.execute("PRAGMA table_info(%s)" % quote(row["name"])).fetchall()

123  if not typerow: continue # for row
