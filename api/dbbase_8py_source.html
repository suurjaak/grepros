24class BaseDataSink(Sink, SqlMixin):
25 """
26 Base class for writing messages to a database.
27
28 Output will have:
29 - table "topics", with topic and message type names
30 - table "types", with message type definitions
31
32 plus:
33 - table "pkg/MsgType" for each topic message type, with detailed fields,
34 BYTEA fields for uint8[], array fields for scalar list attributes,
35 and JSONB fields for lists of ROS messages;
36 with foreign keys if nesting else subtype values as JSON dictionaries;
37 plus underscore-prefixed fields for metadata, like `_topic` as the topic name.
38 If not nesting, only topic message type tables are created.
39 - view "/full/topic/name" for each topic, selecting from the message type table
40
41 If a message type table already exists but for a type with a different MD5 hash,
42 the new table will have its MD5 hash appended to end, as "pkg/MsgType (hash)".
43 """
44

144 self._executemany(sql, self._sql_queue.pop(sql))

202 def _process_topic(self, topic, msg):
203 """
204 Inserts topics-row and creates view `/topic/name` if not already existing.
205
206 Also creates types-row and pkg/MsgType table for this message if not existing.
207 If nesting enabled, creates types recursively.
208 """
209 topickey = api.TypeMeta.make(msg, topic).topickey
210 if topickey in self._checkeds:
211 return
212
213 if topickey not in self._topics:
214 exclude_cols = list(self.MESSAGE_TYPE_TOPICCOLS)
215 if self._nesting_nesting: exclude_cols += self.MESSAGE_TYPE_NESTCOLS
216 tdata = self._make_topic_data(topic, msg, exclude_cols)
217 self._topics[topickey] = tdata
218 self._executescript(tdata["sql"])
219
220 sql, args = self._make_topic_insert_sql(topic, msg)
221 if self.args.VERBOSE:
222 ConsolePrinter.debug("Adding topic %s in %s output.", topic, self.ENGINE)
223 self._topics[topickey]["id"] = self._execute_insert(sql, args)
224
225 if self.COMMIT_INTERVALCOMMIT_INTERVAL: self.db.commit()
226 self._checkeds[topickey] = True

229 def _process_type(self, msg, rootmsg=None):
230 """
231 Creates types-row and pkg/MsgType table if not already existing.
232
233 @return created types-row, or None if already existed
234 """
235 rootmsg = rootmsg or msg
236 with api.TypeMeta.make(msg, root=rootmsg) as m:
237 typename, typekey = (m.typename, m.typekey)
238 if typekey in self._checkeds:
239 return None
240
241 if typekey not in self._types:
242 if self.args.VERBOSE:
243 ConsolePrinter.debug("Adding type %s in %s output.", typename, self.ENGINE)
244 extra_cols = self.MESSAGE_TYPE_TOPICCOLS + self.MESSAGE_TYPE_BASECOLS
245 if self._nesting_nesting: extra_cols += self.MESSAGE_TYPE_NESTCOLS
246 tdata = self._make_type_data(msg, extra_cols)
247 self._schema[typekey] = collections.OrderedDict(tdata.pop("cols"))
248 self._types[typekey] = tdata
249
250 self._executescript(tdata["sql"])
251 sql, args = self._make_type_insert_sql(msg)
252 tdata["id"] = self._execute_insert(sql, args)
253
254
255 nested_tables = self._types[typekey].get("nested_tables") or {}
256 nesteds = api.iter_message_fields(msg, messages_only=True) if self._nesting_nesting else ()
257 for path, submsgs, subtype in nesteds:
258 scalartype = api.scalar(subtype)
259 if subtype == scalartype and "all" != self._nesting_nesting:
260 continue # for path
261
262 subtypehash = not submsgs and self.source.get_message_type_hash(scalartype)
263 if not isinstance(submsgs, (list, tuple)): submsgs = [submsgs]
264 [submsg] = submsgs[:1] or [self.source.get_message_class(scalartype, subtypehash)()]
265 subdata = self._process_type(submsg, rootmsg)
266 if subdata: nested_tables[".".join(path)] = subdata["table_name"]
267 if nested_tables:
268 self._types[typekey]["nested_tables"] = nested_tables
269 sets, where = {"nested_tables": nested_tables}, {"id": self._types[typekey]["id"]}
270 sql, args = self._make_update_sql("types", sets, where)
271 self._cursor.execute(sql, args)
272 self._checkeds[typekey] = True
273 return self._types[typekey]

291 def _populate_type(self, topic, msg, stamp,
292 rootmsg=None, parent_type=None, parent_id=None):
293 """
294 Inserts pkg/MsgType row for message.
295
296 If nesting is enabled, inserts sub-rows for subtypes in message,
297 and returns inserted ID.
298 """
299 rootmsg = rootmsg or msg
300 with api.TypeMeta.make(msg, root=rootmsg) as m:
301 typename, typekey = m.typename, m.typekey
302 with api.TypeMeta.make(rootmsg) as m:
303 topic_id = self._topics[m.topickey]["id"]
304 table_name = self._types[typekey]["table_name"]
305
306 myid = self._get_next_id(table_name) if self._nesting_nesting else None
307 coldefs = self.MESSAGE_TYPE_TOPICCOLS + self.MESSAGE_TYPE_BASECOLS[:-1]
308 colvals = [topic, topic_id, api.to_datetime(stamp), api.to_nsec(stamp)]
309 if self._nesting_nesting:
310 coldefs += self.MESSAGE_TYPE_BASECOLS[-1:] + self.MESSAGE_TYPE_NESTCOLS
311 colvals += [myid, parent_type, parent_id]
312 extra_cols = list(zip([c for c, _ in coldefs], colvals))
313 sql, args = self._make_message_insert_sql(topic, msg, extra_cols)
314 self._ensure_execute(sql, args)
315 if parent_type: self._nested_counts[typekey] = self._nested_counts.get(typekey, 0) + 1
316
317 subids = {} # {message field path: [ids]}
318 nesteds = api.iter_message_fields(msg, messages_only=True) if self._nesting_nesting else ()
319 for subpath, submsgs, subtype in nesteds:
320 scalartype = api.scalar(subtype)
321 if subtype == scalartype and "all" != self._nesting_nesting:
322 continue # for subpath
323 if isinstance(submsgs, (list, tuple)):
324 subids[subpath] = []
325 for submsg in submsgs if isinstance(submsgs, (list, tuple)) else [submsgs]:
326 subid = self._populate_type(topic, submsg, stamp, rootmsg, typename, myid)
327 if isinstance(submsgs, (list, tuple)):
328 subids[subpath].append(subid)
329 if subids:
330 sets, where = {".".join(p): subids[p] for p in subids}, {"_id": myid}
331 sql, args = self._make_update_sql(table_name, sets, where)
332 self._ensure_execute(sql, args)
333 return myid

336 def _ensure_columns(self, cols):
337 """Adds specified columns to any type tables lacking them."""
338 sqls = []
339 for typekey, typecols in self._schema.items():
340 table_name = self._types[typekey]["table_name"]
341 for c, t in ((c, t) for c, t in cols if c not in typecols):
342 sql = "ALTER TABLE %s ADD COLUMN %s %s;" % (quote(table_name), c, t)
343 sqls.append(sql)
344 typecols[c] = t
345 if sqls:
346 self._executescript("\n".join(sqls))
347 self.db.commit()
