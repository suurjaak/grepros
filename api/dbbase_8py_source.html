24 class DataSinkBase(SinkBase, SqlMixin):
25  """
26  Base class for writing messages to a database.
27 
28  Output will have:
29  - table "topics", with topic and message type names
30  - table "types", with message type definitions
31 
32  plus:
33  - table "pkg/MsgType" for each topic message type, with detailed fields,
34  BYTEA fields for uint8[], array fields for scalar list attributes,
35  and JSONB fields for lists of ROS messages;
36  with foreign keys if nesting else subtype values as JSON dictionaries;
37  plus underscore-prefixed fields for metadata, like `_topic` as the topic name.
38  If not nesting, only topic message type tables are created.
39  - view "/full/topic/name" for each topic, selecting from the message type table
40 
41  If a message type table already exists but for a type with a different MD5 hash,
42  the new table will have its MD5 hash appended to end, as "pkg/MsgType (hash)".
43  """

213  def _process_type(self, msg, rootmsg=None):
214  """

216 
217  @return created types-row, or None if already existed
218  """
219  rootmsg = rootmsg or msg
220  with rosapi.TypeMeta.make(msg, root=rootmsg) as m:
221  typename, typekey = (m.typename, m.typekey)
222  if typekey in self._checkeds:
223  return
224

229  if self._nesting: extra_cols += self.MESSAGE_TYPE_NESTCOLS
230  tdata = self._make_type_data(msg, extra_cols)
231  self._schema[typekey] = collections.OrderedDict(tdata.pop("cols"))
232  self._types[typekey] = tdata

236  tdata["id"] = self._execute_insert(sql, args)
237 
238 
239  nested_tables = self._types[typekey].get("nested_tables") or {}
240  nesteds = rosapi.iter_message_fields(msg, messages_only=True) if self._nesting else ()
241  for path, submsgs, subtype in nesteds:
242  scalartype = rosapi.scalar(subtype)
243  if subtype == scalartype and "all" != self._nesting:
244  continue # for path
245

247  if not isinstance(submsgs, (list, tuple)): submsgs = [submsgs]

275  def _populate_type(self, topic, msg, stamp,
276  rootmsg=None, parent_type=None, parent_id=None):
277  """
278  Inserts pkg/MsgType row for message.
279 
280  If nesting is enabled, inserts sub-rows for subtypes in message,
281  and returns inserted ID.
282  """
283  rootmsg = rootmsg or msg
284  with rosapi.TypeMeta.make(msg, root=rootmsg) as m:
285  typename, typekey = m.typename, m.typekey
286  with rosapi.TypeMeta.make(rootmsg) as m:
287  topic_id = self._topics[m.topickey]["id"]
288  table_name = self._types[typekey]["table_name"]
289 
290  myid = self._get_next_id(table_name) if self._nesting else None
291  coldefs = self.MESSAGE_TYPE_TOPICCOLS + self.MESSAGE_TYPE_BASECOLS[:-1]
292  colvals = [topic, topic_id, rosapi.to_datetime(stamp), rosapi.to_nsec(stamp)]
293  if self._nesting:
294  coldefs += self.MESSAGE_TYPE_BASECOLS[-1:] + self.MESSAGE_TYPE_NESTCOLS
295  colvals += [myid, parent_type, parent_id]
296  extra_cols = list(zip([c for c, _ in coldefs], colvals))
297  sql, args = self._make_message_insert_sql(topic, msg, extra_cols)
298  self._ensure_execute(sql, args)
299  if parent_type: self._nested_counts[typekey] = self._nested_counts.get(typekey, 0) + 1
300 
301  subids = {} # {message field path: [ids]}
302  nesteds = rosapi.iter_message_fields(msg, messages_only=True) if self._nesting else ()
303  for subpath, submsgs, subtype in nesteds:
304  scalartype = rosapi.scalar(subtype)
305  if subtype == scalartype and "all" != self._nesting:
306  continue # for subpath
307  if isinstance(submsgs, (list, tuple)):
308  subids[subpath] = []
309  for submsg in submsgs if isinstance(submsgs, (list, tuple)) else [submsgs]:
310  subid = self._populate_type(topic, submsg, stamp, rootmsg, typename, myid)
311  if isinstance(submsgs, (list, tuple)):
312  subids[subpath].append(subid)
313  if subids:
314  sets, where = {".".join(p): subids[p] for p in subids}, {"_id": myid}
315  sql, args = self._make_update_sql(table_name, sets, where)
316  self._ensure_execute(sql, args)
317  return myid
