104  if batch_matched and self._is_max_done():

175  def _parse_patterns(self):
176  """Parses pattern arguments into re.Patterns."""
177  NOBRUTE_SIGILS = r"\A", r"\Z", "?(" # Regex specials ruling out brute precheck
178  BRUTE, FLAGS = not self.args.INVERT, re.DOTALL | (0 if self.args.CASE else re.I)
179  self._patterns.clear()
180  del self._brute_prechecks[:]
181  contents = []
182  for v in self.args.PATTERNS:
183  split = v.find("=", 1, -1)
184  v, path = (v[split + 1:], v[:split]) if split > 0 else (v, ())
185  # Special case if '' or "": add pattern for matching empty string
186  v = (re.escape(v) if self.args.RAW else v) + ("|^$" if v in ("''", '""') else "")
187  path = re.compile(r"(^|\.)%s($|\.)" % ".*".join(map(re.escape, path.split("*")))) \
188  if path else ()
189  contents.append((path, re.compile("(%s)" % v, FLAGS)))
190  if BRUTE and (self.args.RAW or not any(x in v for x in NOBRUTE_SIGILS)):
191  self._brute_prechecks.append(re.compile(v, re.I | re.M))
192  if not self.args.PATTERNS: # Add match-all pattern
193  contents.append(self.ANY_MATCHES[0])
