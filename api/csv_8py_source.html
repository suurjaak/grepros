90  def _make_writer(self, topic, msg):
91  """
92  Returns a csv.writer for writing topic data.
93 
94  File is populated with header if not created during this session.
95  """
96  topickey = rosapi.TypeMeta.make(msg, topic).topickey
97  if not self._lasttopickey:
98  makedirs(os.path.dirname(self._filebase))
99  if self._lasttopickey and topickey != self._lasttopickey:
100  self._files[self._lasttopickey].close() # Avoid hitting ulimit
101  if topickey not in self._files or self._files[topickey].closed:
102  name = self._files[topickey].name if topickey in self._files else None
103  action = "Creating" # Or "Overwriting"
104  if not name:
105  base, ext = os.path.splitext(self._filebase)
106  name = "%s.%s%s" % (base, topic.lstrip("/").replace("/", "__"), ext)
107  if self._overwrite:
108  if os.path.isfile(name) and os.path.getsize(name): action = "Overwriting"
109  open(name, "w").close()
110  else: name = unique_path(name)
111  flags = {"mode": "ab"} if sys.version_info < (3, 0) else {"mode": "a", "newline": ""}
112  f = open(name, **flags)
113  w = CsvWriter(f)
114  if topickey not in self._files:
115  if self.args.VERBOSE:
116  ConsolePrinter.debug("%s %s.", action, name)
117  header = (topic + "/" + ".".join(map(str, p)) for p, _ in self._iter_fields(msg))
118  metaheader = ["__time", "__datetime", "__type"]
119  w.writerow(itertools.chain(metaheader, header))
120  self._files[topickey], self._writers[topickey] = f, w
121  self._lasttopickey = topickey
122  return self._writers[topickey]

124  def _iter_fields(self, msg, top=()):
125  """
126  Yields ((nested, path), scalar value) from ROS message.
127 
128  Lists are returned as ((nested, path, index), value), e.g. (("data", 0), 666).
129  """
130  fieldmap, identity = rosapi.get_message_fields(msg), lambda x: x
131  for k, t in fieldmap.items() if fieldmap != msg else ():
132  v, path, baset = rosapi.get_message_value(msg, k, t), top + (k, ), rosapi.scalar(t)
133  is_sublist = isinstance(v, (list, tuple)) and baset not in rosapi.ROS_BUILTIN_TYPES
134  cast = rosapi.to_sec if baset in rosapi.ROS_TIME_TYPES else identity
135  if isinstance(v, (list, tuple)) and not is_sublist:
136  for i, lv in enumerate(v):
137  yield path + (i, ), cast(lv)
138  elif is_sublist:
139  for i, lmsg in enumerate(v):
140  for lp, lv in self._iter_fields(lmsg, path + (i, )):
141  yield lp, lv
142  elif rosapi.is_ros_message(v, ignore_time=True):
143  for mp, mv in self._iter_fields(v, path):
144  yield mp, mv
145  else:
146  yield path, cast(v)

152  def __init__(self, csvfile, dialect="excel", **fmtparams):
153  """
154  @param csvfile file-like object with `write()` method
155  @param dialect CSV dialect to use, one from `csv.list_dialects()`
156  @param fmtparams override individual format parameters in dialect
157  """
158  self._file = csvfile
159  self._buffer = io.BytesIO() if sys.version_info < (3, 0) else io.StringIO()
160  self._writer = csv.writer(self._buffer, dialect, **dict(fmtparams, lineterminator=""))
161  self._dialect = csv.writer(self._buffer, dialect, **fmtparams).dialect
162  self._format = lambda v: int(v) if isinstance(v, bool) else v
163  if sys.version_info < (3, 0): # Py2, CSV is written in binary mode
164  self._format = lambda v: int(v) if isinstance(v, bool) else \
165  v.encode("utf-8") if isinstance(v, unicode) else v

174  Writes the row to the writer’s file object.

182  """Writes columns to file, returns number of bytes written."""
183  count = self._file.write(inter) if inter else 0
184  self._writer.writerow(cols) # Hack: use csv.writer to format
