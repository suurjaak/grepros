133 def __init__(self, args=None, **kwargs):
134 """
135 @param args arguments as namespace or dictionary, case-insensitive
136 @param args.color False or "never" for not using colors in replacements
137 @param args.highlight highlight matched values (default true)

148 both sides if one value, start and end if more than one,

161 def format_message(self, msg, highlight=False):
162 """Returns message as formatted string, optionally highlighted for matches if configured."""
163 text = self.message_to_yaml(msg).rstrip("\n")
164
165 highlight = highlight and self.args.HIGHLIGHT
166 if self._prefix or self.args.START_LINE or self.args.END_LINE \
167 or self.args.MAX_MESSAGE_LINES or (self.args.LINES_AROUND_MATCH and highlight):
168 lines = text.splitlines()
169
170 if self.args.START_LINE or self.args.END_LINE or self.args.MAX_MESSAGE_LINES:
171 start = self.args.START_LINE or 0
172 start = max(start, -len(lines)) - (start > 0) # <0 to sanity, >0 to 0-base
173 lines = lines[start:start + (self.args.MAX_MESSAGE_LINES or len(lines))]
174 lines = lines and (lines[:-1] + [lines[-1] + self._styles["rst"]])
175
176 if self.args.LINES_AROUND_MATCH and highlight:
177 spans, NUM = [], self.args.LINES_AROUND_MATCH
178 for i, l in enumerate(lines):
179 if MatchMarkers.START in l:
180 spans.append([max(0, i - NUM), min(i + NUM + 1, len(lines))])
181 if MatchMarkers.END in l and spans:
182 spans[-1][1] = min(i + NUM + 1, len(lines))
183 lines = sum((lines[a:b - 1] + [lines[b - 1] + self._styles["rst"]]
184 for a, b in common.merge_spans(spans)), [])
185
186 if self._prefix:
187 lines = [self._prefix + l for l in lines]
188
189 text = "\n".join(lines)
190
191 for a, b in self._format_repls.items() if highlight else ():
192 text = re.sub(r"(%s)\1+" % re.escape(a), r"\1", text) # Remove consecutive duplicates
193 text = text.replace(a, b)
194
195 return text

198 def message_to_yaml(self, val, top=(), typename=None):
199 """Returns ROS message or other value as YAML."""
200 # Refactored from genpy.message.strify_message().
201 unquote = lambda v: v[1:-1] if v[:1] == v[-1:] == '"' else v
202
203 def retag_match_lines(lines):
204 """Adds match tags to lines where wrapping separated start and end."""
205 PH = self._wrapper.placeholder
206 for i, l in enumerate(lines):
207 startpos0, endpos0 = l.find (MatchMarkers.START), l.find (MatchMarkers.END)
208 startpos1, endpos1 = l.rfind(MatchMarkers.START), l.rfind(MatchMarkers.END)
209 if endpos0 >= 0 and (startpos0 < 0 or startpos0 > endpos0):
210 lines[i] = l = re.sub(r"^(\s*)", r"\1" + MatchMarkers.START, l)
211 if startpos1 >= 0 and endpos1 < startpos1 and i + 1 < len(lines):
212 lines[i + 1] = re.sub(r"^(\s*)", r"\1" + MatchMarkers.START, lines[i + 1])
213 if startpos1 >= 0 and startpos1 > endpos1:
214 CUT, EXTRA = (-len(PH), PH) if PH and l.endswith(PH) else (len(l), "")
215 lines[i] = l[:CUT] + MatchMarkers.END + EXTRA
216 return lines
217
218 def truncate(v):
219 """Returns text or list/tuple truncated to length used in final output."""
220 if self.args.LINES_AROUND_MATCH \
221 or (not self.args.MAX_MESSAGE_LINES and (self.args.END_LINE or 0) <= 0): return v
222
223 MAX_CHAR_LEN = 1 + len(MatchMarkers.START) + len(MatchMarkers.END)
224 # For list/tuple, account for comma and space
225 if isinstance(v, (list, tuple)): textlen = bytelen = 2 + len(v) * (2 + MAX_CHAR_LEN)
226 else: textlen, bytelen = self._wrapper.strlen(v), len(v)
227 if textlen < 10000: return v
228
229 # Heuristic optimization: shorten superlong texts before converting to YAML
230 # if outputting a maximum number of lines per message
231 # (e.g. a lidar pointcloud can be 10+MB of text and take 10+ seconds to format).
232 MIN_CHARS_PER_LINE = self._wrapper.width
233 if MAX_CHAR_LEN != 1:
234 MIN_CHARS_PER_LINE = self._wrapper.width // MAX_CHAR_LEN * 2
235 MAX_LINES = self.args.MAX_MESSAGE_LINES or self.args.END_LINE
236 MAX_CHARS = MAX_LEN = MAX_LINES * MIN_CHARS_PER_LINE * self._wrapper.width + 100
237 if bytelen > MAX_CHARS: # Use worst-case max length plus some extra
238 if isinstance(v, (list, tuple)): MAX_LEN = MAX_CHARS // 3
239 v = v[:MAX_LEN]
240 return v
241
242 indent = " " * len(top)
243 if isinstance(val, six.integer_types + (float, bool)):
244 return str(val)
245 if isinstance(val, common.TEXT_TYPES):
246 if val in ("", MatchMarkers.EMPTY):
247 return MatchMarkers.EMPTY_REPL if val else "''"
248 # default_style='"' avoids trailing "...\n"
249 return yaml.safe_dump(truncate(val), default_style='"', width=sys.maxsize).rstrip("\n")
250 if isinstance(val, (list, tuple)):
251 if not val:
252 return "[]"
253 if api.scalar(typename) in api.ROS_STRING_TYPES:
254 yaml_str = yaml.safe_dump(truncate(val)).rstrip('\n')
255 return "\n" + "\n".join(indent + line for line in yaml_str.splitlines())
256 vals = [x for v in truncate(val) for x in [self.message_to_yaml(v, top, typename)] if x]
257 if api.scalar(typename) in api.ROS_NUMERIC_TYPES:
258 return "[%s]" % ", ".join(unquote(str(v)) for v in vals)
259 return ("\n" + "\n".join(indent + "- " + v for v in vals)) if vals else ""
260 if api.is_ros_message(val):
261 MATCHED_ONLY = self.args.MATCHED_FIELDS_ONLY and not self.args.LINES_AROUND_MATCH
262 vals, fieldmap = [], api.get_message_fields(val)
263 prints, noprints = self._patterns["print"], self._patterns["noprint"]
264 fieldmap = api.filter_fields(fieldmap, top, include=prints, exclude=noprints)
265 for k, t in fieldmap.items():
266 v = self.message_to_yaml(api.get_message_value(val, k, t), top + (k, ), t)
267 if not v or MATCHED_ONLY and MatchMarkers.START not in v:
268 continue # for k, t
269
270 if t not in api.ROS_STRING_TYPES: v = unquote(v)
271 if api.scalar(t) in api.ROS_BUILTIN_TYPES:
272 is_strlist = t.endswith("]") and api.scalar(t) in api.ROS_STRING_TYPES
273 is_num = api.scalar(t) in api.ROS_NUMERIC_TYPES
274 extra_indent = indent if is_strlist else " " * len(indent + k + ": ")
275 self._wrapper.reserve_width(self._prefix + extra_indent)
276 self._wrapper.drop_whitespace = t.endswith("]") and not is_strlist
277 self._wrapper.break_long_words = not is_num
278 v = ("\n" + extra_indent).join(retag_match_lines(self._wrapper.wrap(v)))
279 if is_strlist and self._wrapper.strip(v) != "[]": v = "\n" + v
280 vals.append("%s%s: %s" % (indent, k, api.format_message_value(val, k, v)))
281 return ("\n" if indent and vals else "") + "\n".join(vals)
282
283 return str(val)

286 def _configure(self, args):
287 """Initializes output settings."""
288 prints, noprints = args.EMIT_FIELD, args.NOEMIT_FIELD
289 for key, vals in [("print", prints), ("noprint", noprints)]:
290 self._patterns[key] = [(tuple(v.split(".")), common.wildcard_to_regex(v)) for v in vals]
291
292 if args.COLOR not in ("never", False):
293 self._styles.update({"hl0": ConsolePrinter.STYLE_HIGHLIGHT if self.args.HIGHLIGHT
294 else "",
295 "ll0": ConsolePrinter.STYLE_LOWLIGHT,
296 "pfx0": ConsolePrinter.STYLE_SPECIAL, # Content line prefix start
297 "sep0": ConsolePrinter.STYLE_SPECIAL2})
298 self._styles.default_factory = lambda: ConsolePrinter.STYLE_RESET
299
300 WRAPS = args.MATCH_WRAPPER if self.args.HIGHLIGHT else ""
301 if WRAPS is None and args.COLOR in ("never", False): WRAPS = self.NOCOLOR_HIGHLIGHT_WRAPPERS
302 WRAPS = ((WRAPS or [""]) * 2)[:2]
303 self._styles["hl0"] = self._styles["hl0"] + WRAPS[0]
304 self._styles["hl1"] = WRAPS[1] + self._styles["hl1"]
305
306 custom_widths = {MatchMarkers.START: len(WRAPS[0]), MatchMarkers.END: len(WRAPS[1]),
307 self._styles["ll0"]: 0, self._styles["ll1"]: 0,
308 self._styles["pfx0"]: 0, self._styles["pfx1"]: 0,
309 self._styles["sep0"]: 0, self._styles["sep1"]: 0}
310 wrapargs = dict(max_lines=args.MAX_FIELD_LINES,
311 placeholder="%s ...%s" % (self._styles["ll0"], self._styles["ll1"]))
312 if args.WRAP_WIDTH is not None: wrapargs.update(width=args.WRAP_WIDTH)
313 self._wrapper = common.TextWrapper(custom_widths=custom_widths, **wrapargs)
314 self._format_repls = {MatchMarkers.START: self._styles["hl0"],
315 MatchMarkers.END: self._styles["hl1"]}

449 def validate(self):
450 """Returns whether write options are valid and ROS environment set, emits error if not."""
451 if self.validvalid is not None: return self.validvalid
452 result = True
453 if self.args.WRITE_OPTIONS.get("overwrite") not in (None, True, False, "true", "false"):

471 def close(self):
472 """Closes output bag, if any."""
473 if not self._close_printed and self._counts_counts and self._bag:
474 self._close_printed = True
475 self._bag.close()
476 try: sz = common.format_bytes(os.path.getsize(self._bag.filename)
477 if self._bag.filename else self._bag.size or 0)

482 common.plural("message", sum(self._counts_counts.values())),
483 common.plural("topic", self._counts_counts),

488 def _ensure_open(self):
489 """Opens output file if not already open."""
490 if self._bag is not None:
491 self._bag.open()
492 return
493 if common.is_stream(self.args.WRITE):
494 self._bag = api.Bag(self.args.WRITE, mode=getattr(self.args.WRITE, "mode", "w"))
495 return
496 filename = self.args.WRITE
497 if not self._overwrite and os.path.isfile(filename) and os.path.getsize(filename):
498 cls = api.Bag.autodetect(filename)

500 filename = common.unique_path(filename)

509 filename, (" (%s)" % common.format_bytes(sz)) if sz else "")

546 """Publishes message to output topic."""

550 topickey, cls = (m.topickey, m.typeclass)
551 if topickey not in self._pubs:
552 topic2 = self.args.PUBLISH_PREFIX + topic + self.args.PUBLISH_SUFFIX
553 topic2 = self.args.PUBLISH_FIXNAME or topic2

556
557 pub = None
558 if self.args.PUBLISH_FIXNAME:
559 pub = next((v for (_, c), v in self._pubs.items() if c == cls), None)
560 pub = pub or api.create_publisher(topic2, cls, queue_size=self.args.QUEUE_SIZE_OUT)
561 self._pubs[topickey] = pub

577 if self.validvalid is not None: return self.validvalid

579 config_ok = True

589 """Shuts down publishers."""
590 if not self._close_printed and self._counts:

594 common.plural("topic", self._pubs))

641class MultiSink(Sink):
642 """Combines any number of sinks."""

644
645 FLAG_CLASSES = {"PUBLISH": TopicSink, "CONSOLE": ConsoleSink, "APP": AppSink}

648 FORMAT_CLASSES = {"bag": BagSink}

650 def __init__(self, args=None, sinks=(), **kwargs):
651 """
652 Accepts more arguments, given to the real sinks constructed.
653
654 @param args arguments as namespace or dictionary, case-insensitive
655 @param args.console print matches to console
656 @param args.write [[target, format=FORMAT, key=value, ], ]
657 @param args.publish publish matches to live topics
658 @param args.app provide messages to given callback function
659 @param sinks pre-created sinks, arguments will be ignored
660 @param kwargs any and all arguments as keyword overrides, case-insensitive
661 """
662 args = common.ensure_namespace(args, **kwargs)
663 super(MultiSink, self).__init__(args)
664 self.validvalid = True
665
666
667 self.sinks = [cls(args) for flag, cls in self.FLAG_CLASSES.items()
668 if getattr(args, flag, None)] if not sinks else list(sinks)
669
670 for dumpopts in getattr(args, "WRITE", []) if not sinks else ():
671 kwargs = dict(x.split("=", 1) for x in dumpopts[1:] if isinstance(x, common.TEXT_TYPES))
672 kwargs.update(kv for x in dumpopts[1:] if isinstance(x, dict) for kv in x.items())
673 target, cls = dumpopts[0], self.FORMAT_CLASSES.get(kwargs.pop("format", None))
674 if not cls:
675 cls = next((c for c in sorted(self.FORMAT_CLASSES.values(),
676 key=lambda x: x is BagSink)
677 if callable(getattr(c, "autodetect", None))
678 and c.autodetect(target)), None)
679 if not cls:
680 ConsolePrinter.error('Unknown output format in "%s"' % " ".join(map(str, dumpopts)))
681 self.validvalid = False
682 continue # for dumpopts
683 clsargs = common.structcopy(args)
684 clsargs.WRITE, clsargs.WRITE_OPTIONS = target, kwargs
685 self.sinks += [cls(clsargs)]
686
687 def emit_meta(self):
688 """Outputs source metainfo in one sink, if not already emitted."""
689 sink = next((s for s in self.sinks if isinstance(s, ConsoleSink)), None)
690 # Emit meta in one sink only, prefer console
691 sink = sink or self.sinks[0] if self.sinks else None
692 sink and sink.emit_meta()
693
694 def emit(self, topic, msg, stamp=None, match=None, index=None):
695 """Outputs ROS message to all sinks."""
696 stamp, index = self._ensure_stamp_index(topic, msg, stamp, index)
697 for sink in self.sinks:
698 sink.emit(topic, msg, stamp, match, index)
699 super(MultiSink, self).emit(topic, msg, stamp, match, index)
700
701 def bind(self, source):
702 """Attaches source to all sinks, sets thread_excepthook on all sinks."""
703 super(MultiSink, self).bind(source)
704 for sink in self.sinks:
705 sink.bind(source)
706 sink.thread_excepthook = self.thread_excepthook

710 if not self.sinks:
711 ConsolePrinter.error("No output configured.")
712 return bool(self.sinks) and all([sink.validate() for sink in self.sinks]) and self.validvalid
