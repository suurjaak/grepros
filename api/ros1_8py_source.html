63class ROS1Bag(rosbag.Bag, api.BaseBag):
64 """
65 ROS1 bag reader and writer.
66
67 Extends `rosbag.Bag` with more conveniences, smooths over the rosbag bug of ignoring
68 topic and time filters in format v1.2, and smooths over the rosbag bug
69 of yielding messages of wrong type, if message types in different topics
70 have different packages but identical fields and hashes.
71
72 Does **not** smooth over the rosbag bug of writing different types to one topic.
73
74 rosbag does allow writing messages of different types to one topic,
75 just like live ROS topics can have multiple message types published
76 to one topic. And their serialized bytes will actually be in the bag,
77 but rosbag will only register the first type for this topic (unless it is
78 explicitly given another connection header with metadata on the other type).
79
80 All messages yielded will be deserialized by rosbag as that first type,
81 and whether reading will raise an exception or not depends on whether
82 the other type has enough bytes to be deserialized as that first type.
83 """
84

189 def read_messages(self, topics=None, start_time=None, end_time=None,
190 raw=False, connection_filter=None, **__):
191 """
192 Yields messages from the bag, optionally filtered by topic, timestamp and connection details.
193
194 @param topics list of topics or a single topic.
195 If an empty list is given, all topics will be read.
196 @param start_time earliest timestamp of messages to return,
197 as ROS time or convertible (int/float/duration/datetime/decimal)
198 @param end_time latest timestamp of messages to return,
199 as ROS time or convertible (int/float/duration/datetime/decimal)
200 @param connection_filter function to filter connections to include
201 @param raw if true, then returned messages are tuples of
202 (typename, bytes, typehash, typeclass)
203 or (typename, bytes, typehash, position, typeclass),
204 depending on file format
205 @return BagMessage namedtuples of
206 (topic, message, timestamp as rospy.Time)
207 """
208 if self.closedclosedclosedclosed: raise ValueError("I/O operation on closed file.")
209 if "w" == self._mode: raise io.UnsupportedOperation("read")
210
211 hashtypes = {}
212 for n, h in self.__TYPEDEFS: hashtypes.setdefault(h, []).append(n)
213 read_topics = topics if isinstance(topics, (dict, list, set, tuple)) else \
214 [topics] if topics else None
215 dupes = {t: (n, h) for t, n, h in self.__topics
216 if (read_topics is None or t in read_topics) and len(hashtypes.get(h, [])) > 1}
217
218 # Workaround for rosbag.Bag ignoring topic and time filters in format v1.2
219 if self.version != 102 or (not topics and start_time is None and end_time is None):
220 in_range = lambda *_: True
221 else: in_range = lambda t, s: ((not read_topics or t in read_topics) and
222 (start_time is None or s >= start_time) and
223 (end_time is None or s <= end_time))
224
225 start_time, end_time = map(to_time, (start_time, end_time))
226 kwargs = dict(topics=topics, start_time=start_time, end_time=end_time,
227 connection_filter=connection_filter, raw=raw)
228 if not dupes:
229 for topic, msg, stamp in super(ROS1Bag, self).read_messages(**kwargs):
230 if in_range(topic, stamp):
231 TypeMeta.make(msg, topic, self)
232 yield self.BagMessage(topic, msg, stamp)
233 if self.closedclosedclosedclosed: break # for topic
234 return
235
236 for topic, msg, stamp in super(ROS1Bag, self).read_messages(**kwargs):
237 if not in_range(topic, stamp): continue # for topic
238
239 # Workaround for rosbag bug of using wrong type for identical type hashes
240 if topic in dupes:
241 typename, typehash = (msg[0], msg[2]) if raw else (msg._type, msg._md5sum)
242 if dupes[topic] != (typename, typehash):
243 if raw:
244 msg = msg[:-1] + (self.get_message_classget_message_class(typename, typehash), )
245 else:
246 msg = self._convert_message(msg, *dupes[topic])
247 TypeMeta.make(msg, topic, self)
248 yield self.BagMessage(topic, msg, stamp)
249 if self.closedclosedclosedclosed: break # for topic

323 def _populate_meta(self):
324 """Populates topics and message type definitions and hashes."""
325 result = collections.Counter() # {(topic, typename, typehash): count}
326 counts = collections.Counter() # {connection ID: count}
327 for c in self._chunks:
328 for c_id, count in c.connection_counts.items():
329 counts[c_id] += count
330 for c in self._connections.values():
331 result[(c.topic, c.datatype, c.md5sum)] += counts[c.id]
332 self.__TYPEDEFS[(c.datatype, c.md5sum)] = c.msg_def

336 def _register_write(self, topic, msg, raw=False, connection_header=None):
337 """Registers message in local metadata, writes connection header if new type for topic."""
338 if raw: typename, typehash, typeclass = msg[0], msg[2], msg[3]
339 else: typename, typehash, typeclass = msg._type, msg._md5sum, type(msg)
340 topickey, typekey = (topic, typename, typehash), (typename, typehash)
341
342 if topickey not in self.__topics \
343 and any(topic == t and typekey != (n, h) for t, n, h in self.__topics):
344 # Write connection header if topic already present but with different type
345 if not connection_header:
346 connection_header = {"topic": topic, "type": typename, "md5sum": typehash,
347 "message_definition": typeclass._full_text}
348 conn_id = len(self._connections)
349 connection_info = rosbag.bag._ConnectionInfo(conn_id, topic, connection_header)
350 self._write_connection_record(connection_info, encrypt=False)
351 self._connections[conn_id] = self._topic_connections[topic] = connection_info
352
353 self.__TYPES.setdefault(typekey, typeclass)
354 self.__TYPEDEFS.setdefault(typekey, typeclass._full_text)
355 self.__topics[topickey] = self.__topics.get(topickey, 0) + 1

358 def _ensure_typedef(self, typename, typehash=None):
359 """Parses subtype definition from any full definition where available, if not loaded."""
360 typehash = typehash or next((h for n, h in self.__TYPEDEFS if n == typename), None)
361 typekey = (typename, typehash)
362 if typekey not in self.__TYPEDEFS:
363 for (roottype, roothash), rootdef in list(self.__TYPEDEFS.items()):
364 rootkey = (roottype, roothash)
365 if self.__PARSEDS.get(rootkey): continue # for (roottype, roothash)
366
367 subdefs = tuple(parse_definition_subtypes(rootdef).items()) # ((typename, typedef), )
368 subhashes = {n: calculate_definition_hash(n, d, subdefs) for n, d in subdefs}
369 self.__TYPEDEFS.update(((n, subhashes[n]), d) for n, d in subdefs)
370 self.__PARSEDS.update(((n, h), True) for n, h in subhashes.items())
371 self.__PARSEDS[rootkey] = True
372 if typekey in self.__TYPEDEFS:
373 break # for (roottype, roothash)
374 self.__TYPEDEFS.setdefault(typekey, "")

378 def reindex_file(f, progress, *args, **kwargs):
379 """
380 Reindexes bag, making a backup copy in file directory.
381
382 @param progress show progress bar for reindexing status
383 """
384 KWS = ["mode", "compression", "chunk_threshold",
385 "allow_unindexed", "options", "skip_index"]
386 kwargs.update(zip(KWS, args), allow_unindexed=True)
387 copied, bar, f2 = False, None, None
388 if progress:
389 fmt = lambda s: common.format_bytes(s, strip=False)
390 name, size = os.path.basename(f), os.path.getsize(f)
391 aftertemplate = " Reindexing %s (%s): {afterword}" % (name, fmt(size))
392 bar = common.ProgressBar(size, interval=0.1, pulse=True, aftertemplate=aftertemplate)
393
394 ConsolePrinter.warn("Unindexed bag %s, reindexing.", f)
395 bar and bar.update(0).start() # Start progress pulse
396 try:
397 with rosbag.Bag(f, **kwargs) as inbag:
398 inplace = (inbag.version > 102)
399
400 f2 = "%s.orig%s" % os.path.splitext(f)
401 shutil.copy(f, f2)
402 copied = True
403
404 inbag, outbag = None, None
405 outkwargs = dict(kwargs, mode="a" if inplace else "w", allow_unindexed=True)
406 try:
407 inbag = rosbag.Bag(f2, **dict(kwargs, mode="r"))
408 outbag = rosbag.Bag(f, **outkwargs)
409 Bag._run_reindex(inbag, outbag, bar)
410 bar and bar.update(bar.max)
411 except BaseException: # Ensure steady state on KeyboardInterrupt/SystemExit
412 inbag and inbag.close()
413 outbag and outbag.close()
414 copied and shutil.move(f2, f)
415 raise
416 finally: bar and bar.update(bar.value, flush=True).stop()
417 inbag.close()
418 outbag.close()

422 def _run_reindex(inbag, outbag, bar=None):
423 """Runs reindexing, showing optional progress bar."""
424 update_bar = noop = lambda s: None
425 indexbag, writebag = (inbag, outbag) if inbag.version == 102 else (outbag, None)
426 if bar:
427 fmt = lambda s: common.format_bytes(s, strip=False)
428 update_bar = lambda s: (setattr(bar, "afterword", fmt(s)),
429 setattr(bar, "pulse", False), bar.update(s).stop())
430 # v102: build index from inbag, write all messages to outbag.
431 # Other versions: re-build index in outbag file in-place.
432 progress = update_bar if not writebag else noop # Incremental progress during re-build
433 for offset in indexbag.reindex():
434 progress(offset)

436 return

446def init_node(name):
447 """
448 Initializes a ROS1 node if not already initialized.
449
450 Blocks until ROS master available.
451 """
452 if not validate(live=True):
453 return
454

460 try: uri = master.getUri()[-1]

464 "Unable to register with master. Will keep trying.")

468 ConsolePrinter.debug("Connected to ROS master at %s.", uri)
469 available = True
470 try: rospy.get_rostime()
471 except Exception: # Init node only if not already inited

527def create_subscriber(topic, typename, handler, queue_size):
528 """
529 Returns a rospy.Subscriber.
530
531 Local message packages are not required. Subscribes as AnyMsg,
532 creates message class dynamically from connection info,
533 and deserializes message before providing to handler.
534
535 Supplemented with .get_message_class(), .get_message_definition(),
536 .get_message_type_hash(), and .get_qoses().
537
538 The supplementary .get_message_xyz() methods should only be invoked after at least one message
539 has been received from the topic, as they get populated from live connection metadata.
540 """
541 def myhandler(msg):
542 if msg._connection_header["type"] != typename:
543 return
544 typekey = (typename, msg._connection_header["md5sum"])
545 if typename not in TYPECLASSES:
546 typedef = msg._connection_header["message_definition"]
547 for name, cls in genpy.dynamic.generate_dynamic(typename, typedef).items():
548 TYPECLASSES.setdefault((name, cls._md5sum), cls)
549 handler(TYPECLASSES[typekey]().deserialize(msg._buff))
550
551 sub = rospy.Subscriber(topic, rospy.AnyMsg, myhandler, queue_size=queue_size)
552 sub.get_message_class = lambda: next(c for (n, h), c in TYPECLASSES.items()
553 if n == typename)
554 sub.get_message_definition = lambda: next(get_message_definition(c)
555 for (n, h), c in TYPECLASSES.items() if n == typename)
556 sub.get_message_type_hash = lambda: next(h for n, h in TYPECLASSES if n == typename)
557 sub.get_qoses = lambda: None
558 return sub

734def to_time(val):
735 """Returns value as ROS1 time if convertible (int/float/duration/datetime/decimal), else value."""
736 result = val
737 if isinstance(val, decimal.Decimal):
738 result = rospy.Time(int(val), float(val % 1) * 10**9)
739 elif isinstance(val, datetime.datetime):
740 result = rospy.Time(int(val.timestamp()), 1000 * val.microsecond)
741 elif isinstance(val, six.integer_types + (float, )):

744 result = rospy.Time(val.secs, val.nsecs)
