357 def create_subscriber(topic, typename, handler, queue_size):
358  """
359  Returns a rospy.Subscriber.
360 
361  Local message packages are not required. Subscribes as AnyMsg,
362  creates message class dynamically from connection info,
363  and deserializes message before providing to handler.
364 
365  Supplemented with .get_message_class(), .get_message_definition(),
366  .get_message_type_hash(), and .get_qoses().
367 
368  The supplementary .get_message_xyz() methods should only be invoked after at least one message
369  has been received from the topic, as they get populated from live connection metadata.
370  """
371  def myhandler(msg):
372  if msg._connection_header["type"] != typename:
373  return
374  typekey = (typename, msg._connection_header["md5sum"])
375  if typename not in TYPECLASSES:
376  typedef = msg._connection_header["message_definition"]
377  for name, cls in genpy.dynamic.generate_dynamic(typename, typedef).items():
378  TYPECLASSES.setdefault((name, cls._md5sum), cls)
379  handler(TYPECLASSES[typekey]().deserialize(msg._buff))
380 
381  sub = rospy.Subscriber(topic, rospy.AnyMsg, myhandler, queue_size=queue_size)
382  sub.get_message_class = lambda: next(c for (n, h), c in TYPECLASSES.items()
383  if n == typename)
384  sub.get_message_definition = lambda: next(get_message_definition(c)
385  for (n, h), c in TYPECLASSES.items() if n == typename)
386  sub.get_message_type_hash = lambda: next(h for n, h in TYPECLASSES if n == typename)
387  sub.get_qoses = lambda: None
388  return sub
