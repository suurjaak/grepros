31 class PostgresSink(DataSinkBase):
32  """
33  Writes messages to a Postgres database.
34 
35  Output will have:
36  - table "topics", with topic and message type names
37  - table "types", with message type definitions
38 
39  plus:
40  - table "pkg/MsgType" for each topic message type, with detailed fields,
41  BYTEA fields for uint8[], array fields for scalar list attributes,
42  and JSONB fields for lists of ROS messages;
43  with foreign keys if nesting else subtype values as JSON dictionaries;
44  plus underscore-prefixed fields for metadata, like `_topic` as the topic name.
45  If not nesting, only topic message type tables are created.
46  - view "/full/topic/name" for each topic, selecting from the message type table
47 
48  If a message type table already exists but for a type with a different MD5 hash,
49  the new table will have its MD5 hash appended to end, as "pkg/MsgType (hash)".
50  """

148  def _get_next_id(self, table):
149  """Returns next cached ID value, re-populating empty cache from sequence."""
150  if not self._id_queue.get(table):
151  MAXLEN = self._get_dialect_option("maxlen_entity")
152  seqbase, seqsuffix = table, "_%s_seq" % self.MESSAGE_TYPE_BASECOLS[-1][0]
153  if MAXLEN: seqbase = seqbase[:MAXLEN - len(seqsuffix)]
154  sql = "SELECT nextval('%s') AS id" % quote(seqbase + seqsuffix)
155  for _ in range(self.ID_SEQUENCE_STEP):
156  self._cursor.execute(sql)
157  self._id_queue[table].append(self._cursor.fetchone()["id"])
158  return self._id_queue[table].popleft()

161  def _make_column_value(self, value, typename=None):
162  """Returns column value suitable for inserting to database."""
163  TYPES = self._get_dialect_option("types")
164  plaintype = rosapi.scalar(typename) # "string<=10" -> "string"
165  v = value
166  # Common in JSON but disallowed in Postgres
167  replace = {float("inf"): None, float("-inf"): None, float("nan"): None}
168  if not typename:
169  v = psycopg2.extras.Json(v, json.dumps)
170  elif isinstance(v, (list, tuple)):
171  scalartype = rosapi.scalar(typename)
172  if scalartype in rosapi.ROS_TIME_TYPES:
173  v = [self._convert_time_value(x, scalartype) for x in v]
174  elif scalartype not in rosapi.ROS_BUILTIN_TYPES:
175  if self._nesting: v = None
176  else: v = psycopg2.extras.Json([rosapi.message_to_dict(m, replace)
177  for m in v], json.dumps)
178  elif "BYTEA" in (TYPES.get(typename),
179  TYPES.get(rosapi.canonical(typename, unbounded=True))):
180  v = psycopg2.Binary(bytes(bytearray(v))) # Py2/Py3 compatible
181  else:
182  v = list(self._convert_column_value(v, typename)) # Ensure not-tuple for psycopg2
183  elif rosapi.is_ros_time(v):
184  v = self._convert_time_value(v, typename)
185  elif plaintype not in rosapi.ROS_BUILTIN_TYPES:
186  v = psycopg2.extras.Json(rosapi.message_to_dict(v, replace), json.dumps)
187  else:
188  v = self._convert_column_value(v, plaintype)
189  return v
