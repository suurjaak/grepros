169 class ConditionMixin(object):
170  """
171  Provides topic conditions evaluation.
172 
173  Evaluates a set of Python expressions, with a namespace of:
174  - msg: current message being checked
175  - topic: current topic being read
176  - <topic /any/name> messages in named or wildcarded topic
177 
178  <topic ..> gets replaced with an object with the following behavior:
179  - len(obj) -> number of messages processed in topic
180  - bool(obj) -> whether there are any messages in topic
181  - obj[pos] -> topic message at position (from latest if negative, first if positive)
182  - obj.x -> attribute x of last message
183 
184  All conditions need to evaluate as true for a message to be processable.
185  If a condition tries to access attributes of a message not yet present,
186  condition evaluates as false.
187 
188  If a condition topic matches more than one real topic (by wildcard or by
189  different types in one topic), evaluation is done for each set of
190  topics separately, condition passing if any set passes.
191 
192  Example condition: `<topic */control_enable>.data and <topic */cmd_vel>.linear.x > 0`
193  `and <topic */cmd_vel>.angular.z < 0.02`.
194  """
195

258  def is_processable(self, topic, index, stamp, msg):
259  """Returns whether current state passes conditions, if any."""
260  result = True
261  if not self._conditions:
262  return result
263  for i, (expr, code) in enumerate(self._conditions.items()):
264  topics = self._topics_per_condition[i]
265  wildcarded = [t for t in topics if t in self._wildcard_topics]
266  realcarded = {wt: [(t, n, h) for (t, n, h) in self._lastmsgs if p.match(t)]
267  for wt in wildcarded for p in [self._wildcard_topics[wt]]}
268  variants = [[(wt, (t, n, h)) for (t, n, h) in tt] or [(wt, (wt, None))]

270  variants = variants or [[None]] # Ensure one iteration if no wildcards to combine

272  result = False

274  if remaps == (None, ): remaps = ()
275  getter = functools.partial(self._get_topic_instance, remap=dict(remaps))
276  ns = {"topic": topic, "msg": msg, "get_topic": getter}
277  try: result = eval(code, ns)
278  except self.NoMessageException: pass
279  except Exception as e:
280  ConsolePrinter.error('Error evaluating condition "%s": %s', expr, e)
281  raise
282  if result: break # for remaps
283  if not result: break # for i,
284  return result

343  """Parses condition expressions and populates local structures."""
