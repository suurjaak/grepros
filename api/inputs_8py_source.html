193class ConditionMixin(object):
194 """
195 Provides topic conditions evaluation.
196
197 Evaluates a set of Python expressions, with a namespace of:
198 - msg: current message being checked
199 - topic: current topic being read
200 - <topic /any/name> messages in named or wildcarded topic
201
202 <topic ..> gets replaced with an object with the following behavior:
203 - len(obj) -> number of messages processed in topic
204 - bool(obj) -> whether there are any messages in topic
205 - obj[pos] -> topic message at position (from latest if negative, first if positive)
206 - obj.x -> attribute x of last message
207
208 All conditions need to evaluate as true for a message to be processable.
209 If a condition tries to access attributes of a message not yet present,
210 condition evaluates as false.
211
212 If a condition topic matches more than one real topic (by wildcard or by
213 different types in one topic), evaluation is done for each set of
214 topics separately, condition passing if any set passes.
215
216 Example condition: `<topic */control_enable>.data and <topic */cmd_vel>.linear.x > 0`
217 `and <topic */cmd_vel>.angular.z < 0.02`.
218 """
219
220 TOPIC_RGX = re.compile(r"<topic\s+([^\s><]+)\s*>") # "<topic /some/thing>"
221

223 DEFAULT_ARGS = dict(CONDITION=())

230 Object for <topic x> replacements in condition expressions.

232 - len(topic) -> number of messages processed in topic
233 - bool(topic) -> whether there are any messages in topic
234 - topic[x] -> history at -1 -2 for last and but one, or 0 1 for first and second
235 - topic.x -> attribute x of last message
236 """

286 def is_processable(self, topic, msg, stamp, index=None):
287 """Returns whether message passes passes current state conditions, if any."""
288 result = True
289 if not self._conditions:
290 return result
291 for i, (expr, code) in enumerate(self._conditions.items()):
292 topics = self._topics_per_condition[i]
293 wildcarded = [t for t in topics if t in self._wildcard_topics]
294 realcarded = {wt: [(t, n, h) for (t, n, h) in self._lastmsgs if p.match(t)]
295 for wt in wildcarded for p in [self._wildcard_topics[wt]]}
296 variants = [[(wt, (t, n, h)) for (t, n, h) in tt] or [(wt, (wt, None))]
297 for wt, tt in realcarded.items()]
298 variants = variants or [[None]] # Ensure one iteration if no wildcards to combine
299
300 result = False
301 for remaps in itertools.product(*variants): # [(wildcard1, realname1), (wildcard2, ..]
302 if remaps == (None, ): remaps = ()
303 getter = functools.partial(self._get_topic_instance, remap=dict(remaps))
304 ns = {"topic": topic, "msg": msg, "get_topic": getter}
305 try: result = eval(code, ns)
306 except self.NoMessageException: pass
307 except Exception as e:
308 ConsolePrinter.error('Error evaluating condition "%s": %s', expr, e)
309 raise
310 if result: break # for remaps
311 if not result: break # for i,
312 return result

327 def is_conditions_topic(self, topic, pure=True):
328 """

345 def conditions_register_message(self, topic, msg):
346 """Retains message for condition evaluation if in condition topic."""
347 if self.is_conditions_topic(topic, pure=False):
348 topickey = api.TypeMeta.make(msg, topic).topickey
349 self._lastmsgs[topickey].append(msg)
350 if len(self._lastmsgs[topickey]) > self._topic_limits[topic][-1]:
351 self._lastmsgs[topickey].popleft()
352 if len(self._firstmsgs[topickey]) < self._topic_limits[topic][0]:
353 self._firstmsgs[topickey].append(msg)

355 def _get_topic_instance(self, topic, remap=None):
356 """
357 Returns Topic() by name.
358
359 @param remap optional remap dictionary as {topic1: (topic2, typename, typehash)}
360 """
361 if remap and topic in remap:
362 topickey = remap[topic]
363 else:
364 topickey = next(((t, n, h) for (t, n, h) in self._lastmsgs if t == topic), None)
365 if topickey not in self._counts:
366 return self.Empty()
367 c, f, l = (d[topickey] for d in (self._counts, self._firstmsgs, self._lastmsgs))
368 return self.Topic(c, f, l)

370 def _configure_conditions(self, args):
371 """Parses condition expressions and populates local structures."""
372 for v in args.CONDITION:
373 topics = list(set(self.TOPIC_RGX.findall(v)))
374 self._topic_states.update({t: True for t in topics})
375 self._topics_per_condition.append(topics)
376 for t in (t for t in topics if "*" in t):
377 self._wildcard_topics[t] = common.wildcard_to_regex(t, end=True)
378 expr = self.TOPIC_RGX.sub(r'get_topic("\1")', v)
379 self._conditions[expr] = compile(expr, "", "eval")
380
381 for v in args.CONDITION: # Set history length from <topic x>[index]
382 indexexprs = re.findall(self.TOPIC_RGX.pattern + r"\s*\[([^\]]+)\]", v)
383 for topic, indexexpr in indexexprs:
384 limits = self._topic_limits[topic]
385 try:
386 index = eval(indexexpr) # If integer, set history limits
387 limits[index < 0] = max(limits[index < 0], abs(index) + (index >= 0))
388 except Exception: continue # for topic
