432 def validate_args(args):
433  """
434  Validates arguments, prints errors, returns success.
435 
436  @param args arguments object like argparse.Namespace
437  """
438  errors = collections.defaultdict(list) # {category: [error, ]}
439 
440  # Validate --write .. key=value
441  for opts in args.WRITE: # List of lists, one for each --write
442  erropts = []
443  for opt in opts[1:]:
444  try: dict([opt.split("=", 1)])
445  except Exception: erropts.append(opt)
446  if erropts:
447  errors[""].append('Invalid KEY=VALUE in "--write %s": %s' %
448  (" ".join(opts), " ".join(erropts)))
449 
450  for n, v in [("START_TIME", args.START_TIME), ("END_TIME", args.END_TIME)]:
451  if v is None: continue # for v, n
452  try: v = float(v)
453  except Exception: pass
454  try: not isinstance(v, float) and setattr(args, n, parse_datetime(v))
455  except Exception: errors[""].append("Invalid ISO datetime for %s: %s" %
456  (n.lower().replace("_", " "), v))
457 
458  for v in args.PATTERNS if not args.RAW else ():
459  split = v.find("=", 1, -1) # May be "PATTERN" or "attribute=PATTERN"
460  v = v[split + 1:] if split > 0 else v
461  try: re.compile(re.escape(v) if args.RAW else v)
462  except Exception as e:
463  errors["Invalid regular expression"].append("'%s': %s" % (v, e))
464 
465  for v in args.CONDITIONS:
466  v = inputs.ConditionMixin.TOPIC_RGX.sub("dummy", v)
467  try: compile(v, "", "eval")
468  except SyntaxError as e:
469  errors["Invalid condition"].append("'%s': %s at %schar %s" %
470  (v, e.msg, "line %s " % e.lineno if e.lineno > 1 else "", e.offset))
471  except Exception as e:
472  errors["Invalid condition"].append("'%s': %s" % (v, e))
473 
474  for err in errors.get("", []):
475  ConsolePrinter.error(err)
476  for category in filter(bool, errors):
477  ConsolePrinter.error(category)
478  for err in errors[category]:
