398def make_parser():
399 """Returns a configured ArgumentParser instance."""
400 kws = dict(description=ARGUMENTS["description"], epilog=ARGUMENTS["epilog"],
401 formatter_class=HelpFormatter, add_help=False)
402 argparser = argparse.ArgumentParser(**kws)
403 for arg in map(dict, ARGUMENTS["arguments"]):
404 argparser.add_argument(*arg.pop("args"), **arg)
405 for group, groupargs in ARGUMENTS.get("groups", {}).items():
406 grouper = argparser.add_argument_group(group)
407 for arg in map(dict, groupargs):
408 grouper.add_argument(*arg.pop("args"), **arg)
409 return argparser

412def process_args(args):
413 """
414 Converts or combines arguments where necessary, returns full args.
415
416 @param args arguments object like argparse.Namespace
417 """
418 for arg in sum(ARGUMENTS.get("groups", {}).values(), ARGUMENTS["arguments"][:]):
419 name = arg.get("dest") or arg["args"][0]
420 if "version" != arg.get("action") and argparse.SUPPRESS != arg.get("default") \
421 and "HELP" != name and not hasattr(args, name):
422 value = False if arg.get("store_true") else True if arg.get("store_false") else None
423 setattr(args, name, arg.get("default", value))
424
425 if args.CONTEXT:
426 args.BEFORE = args.AFTER = args.CONTEXT
427
428 # Default to printing metadata for publish/write if no console output
429 args.VERBOSE = False if args.SKIP_VERBOSE else \
430 (args.VERBOSE or not args.CONSOLE and bool(CLI_ARGS))
431
432 # Show progress bar only if no console output
433 args.PROGRESS = args.PROGRESS and not args.CONSOLE
434
435 # Print filename prefix on each console message line if not single specific file
436 args.LINE_PREFIX = args.LINE_PREFIX and (args.RECURSE or len(args.FILE) != 1
437 or args.PATH or any("*" in x for x in args.FILE))
438
439 for k, v in vars(args).items(): # Flatten lists of lists and drop duplicates
440 if k != "WRITE" and isinstance(v, list):
441 here = set()
442 setattr(args, k, [x for xx in v for x in (xx if isinstance(xx, list) else [xx])
443 if not (x in here or here.add(x))])
444
445 for n, v in [("START_TIME", args.START_TIME), ("END_TIME", args.END_TIME)]:
446 if not isinstance(v, (six.binary_type, six.text_type)): continue # for v, n
447 try: v = float(v)
448 except Exception: pass # If numeric, leave as string for source to process as relative time
449 try: not isinstance(v, float) and setattr(args, n, parse_datetime(v))
450 except Exception: pass
451
452 return args

455def validate_args(args):
456 """
457 Validates arguments, prints errors, returns success.
458
459 @param args arguments object like argparse.Namespace
460 """
461 errors = collections.defaultdict(list) # {category: [error, ]}
462
463 # Validate --write .. key=value
464 for opts in args.WRITE: # List of lists, one for each --write
465 erropts = []
466 for opt in opts[1:]:
467 try: dict([opt.split("=", 1)])
468 except Exception: erropts.append(opt)
469 if erropts:
470 errors[""].append('Invalid KEY=VALUE in "--write %s": %s' %
471 (" ".join(opts), " ".join(erropts)))
472
473 for n, v in [("START_TIME", args.START_TIME), ("END_TIME", args.END_TIME)]:
474 if v is None: continue # for v, n
475 try: v = float(v)
476 except Exception: pass
477 try: not isinstance(v, float) and setattr(args, n, parse_datetime(v))
478 except Exception: errors[""].append("Invalid ISO datetime for %s: %s" %
479 (n.lower().replace("_", " "), v))
480
481 for v in args.PATTERN if not args.FIXED_STRING else ():
482 split = v.find("=", 1, -1) # May be "PATTERN" or "attribute=PATTERN"
483 v = v[split + 1:] if split > 0 else v
484 try: re.compile(re.escape(v) if args.FIXED_STRING else v)
485 except Exception as e:
486 errors["Invalid regular expression"].append("'%s': %s" % (v, e))
487
488 for v in args.CONDITION:
489 v = inputs.ConditionMixin.TOPIC_RGX.sub("dummy", v)
490 try: compile(v, "", "eval")
491 except SyntaxError as e:
492 errors["Invalid condition"].append("'%s': %s at %schar %s" %
493 (v, e.msg, "line %s " % e.lineno if e.lineno > 1 else "", e.offset))
494 except Exception as e:
495 errors["Invalid condition"].append("'%s': %s" % (v, e))
496
497 for err in errors.get("", []):
498 ConsolePrinter.log(logging.ERROR, err)
499 for category in filter(bool, errors):
500 ConsolePrinter.log(logging.ERROR, category)
501 for err in errors[category]:
502 ConsolePrinter.log(logging.ERROR, " %s" % err)
503 return not errors

527def run():
528 """Parses command-line arguments and runs search."""
529 global CLI_ARGS
530 CLI_ARGS = sys.argv[1:]
531 MatchMarkers.populate("%08x" % random.randint(1, 1E9))
532 preload_plugins()
533 argparser = make_parser()
534 if not CLI_ARGS:
535 argparser.print_usage()
536 return
537
538 atexit.register(flush_stdout)
539 args, _ = argparser.parse_known_args(CLI_ARGS)
540 if args.HELP:
541 argparser.print_help()
542 return
543
544 BREAK_EXS = (KeyboardInterrupt, )
545 try: BREAK_EXS += (BrokenPipeError, ) # Py3
546 except NameError: pass # Py2
547
548 source, sink = None, None
549 try:
550 ConsolePrinter.configure({"always": True, "never": False}.get(args.COLOR))
551 if not validate_args(process_args(args)):
552 sys.exit(1)
553
554 source = plugins.load("source", args) or \
555 (inputs.TopicSource if args.LIVE else inputs.BagSource)(args)
556 if not source.validate():
557 sys.exit(1)
558 sink = outputs.MultiSink(args)
559 sink.sinks.extend(filter(bool, plugins.load("sink", args, collect=True)))
560 if not sink.validate():
561 sys.exit(1)
562
563 thread_excepthook = lambda t, e: (ConsolePrinter.error(t), sys.exit(1))
564 source.thread_excepthook = sink.thread_excepthook = thread_excepthook
565 grepper = plugins.load("scan", args) or search.Scanner(args)
566 grepper.work(source, sink)
567 except BREAK_EXS:
568 try: source and source.close()
569 except (Exception, KeyboardInterrupt): pass
570 try: sink and sink.close()
571 except (Exception, KeyboardInterrupt): pass
572 # Redirect remaining output to devnull to avoid another BrokenPipeError
573 try: os.dup2(os.open(os.devnull, os.O_WRONLY), sys.stdout.fileno())
574 except (Exception, KeyboardInterrupt): pass
575 sys.exit()
576 finally:
577 sink and sink.close()
578 source and source.close()
579 api.shutdown_node()
