159  def _process_type(self, msg, rootmsg=None):
160  """
161  Builds and writes CREATE TABLE statement for message type if not already built.
162 
163  Builds statements recursively for nested types if configured.
164 
165  @return built SQL, or None if already built
166  """
167  rootmsg = rootmsg or msg
168  typekey = rosapi.TypeMeta.make(msg, root=rootmsg).typekey
169  if typekey in self._types:
170  return None
171 
172  extra_cols = [(c, self._make_column_type(t, fallback="int64" if "time" == t else None))
173  for c, t in self.MESSAGE_TYPE_BASECOLS]
174  self._types[typekey] = self._make_type_data(msg, extra_cols, rootmsg)
175  self._schema[typekey] = collections.OrderedDict(self._types[typekey].pop("cols"))
176 
177  self._write_entity("table", self._types[typekey])
178  if self._nesting: self._process_nested(msg, rootmsg)
179  return self._types[typekey]["sql"]

182  def _process_nested(self, msg, rootmsg):
183  """Builds anr writes CREATE TABLE statements for nested types."""
184  nesteds = rosapi.iter_message_fields(msg, messages_only=True) if self._nesting else ()
185  for path, submsgs, subtype in nesteds:
186  scalartype = rosapi.scalar(subtype)
187  if subtype == scalartype and "all" != self._nesting: continue # for path
188 
189  subtypehash = self.source.get_message_type_hash(scalartype)
190  subtypekey = (scalartype, subtypehash)
191  if subtypekey in self._types: continue # for path
192 
193  if not isinstance(submsgs, (list, tuple)): submsgs = [submsgs]
194  [submsg] = submsgs[:1] or [self.source.get_message_class(scalartype, subtypehash)()]
195  subsql = self._process_type(submsg, rootmsg)
196  if subsql: self._nested_types[subtypekey] = subsql

217  def _write_entity(self, category, item):
218  """Writes table or view SQL statement to file."""

220  if "table" == category:
