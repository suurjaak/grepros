62  def _validate_dialect_file(self):
63  """Returns whether "dialect-file" is valid in args.WRITE_OPTIONS."""
64  ok = True
65  if self._args.WRITE_OPTIONS.get("dialect-file"):
66  filename = self._args.WRITE_OPTIONS["dialect-file"]
67  try:
68  with open(filename) as f:
69  dialects = yaml.safe_load(f.read())
70  if any(not isinstance(v, dict) for v in dialects.values()):
71  raise Exception("Each dialect must be a dictionary.")
72  for opts in dialects.values():
73  for k, v in list(opts.get("adapters", {}).items()):
74  try: opts["adapters"][k] = import_item(v)
75  except ImportError:
76  ok = False
77  ConsolePrinter.error("Error loading adapter %r for %r "
78  "in SQL dialect file %r.", v, k, filename)
79  merge_dicts(self.DIALECTS, dialects)
80  except Exception as e:
81  ok = False
82  ConsolePrinter.error("Error reading SQL dialect file %r: %s", filename, e)
83 
84  # Populate ROS type aliases like "byte" and "char"
85  for opts in self.DIALECTS.values() if ok else ():
86  for rostype in list(opts.get("types", {})):
87  alias = rosapi.get_type_alias(rostype)
88  if alias:
89  opts["types"][alias] = opts["types"][rostype]
90  if alias and rostype + "[]" in opts["types"]:
91  opts["types"][alias + "[]"] = opts["types"][rostype + "[]"]
92 
93  return ok

96  def _validate_dialect(self):
97  """Returns whether "dialect" is valid in args.WRITE_OPTIONS."""
98  ok = True
99  if "dialect" in self._args.WRITE_OPTIONS \
100  and self._args.WRITE_OPTIONS["dialect"] not in tuple(filter(bool, self.DIALECTS)):

106  return ok

117  def _make_topic_data(self, topic, msg, exclude_cols=()):
118  """
119  Returns full data dictionary for topic, including view name and SQL.
120 
121  @param exclude_cols list of column names to exclude from view SELECT, if any
122  @return {"name": topic name, "type": message type name as "pkg/Cls",
123  "table_name": message type table name, "view_name": topic view name,
124  "md5": message type definition MD5 hash, "sql": "CREATE VIEW .."}
125  """
126  with rosapi.TypeMeta.make(msg, topic) as m:
127  typename, typehash, typekey = (m.typename, m.typehash, m.typekey)
128 
129  table_name = self._types[typekey]["table_name"]
130  pkgname, clsname = typename.split("/", 1)
131  nameargs = {"topic": topic, "type": typename, "hash": typehash,
132  "package": pkgname, "class": clsname}
133  view_name = self._make_entity_name("view", nameargs)
134 
135  sqlargs = dict(nameargs, view=quote(view_name), table=quote(table_name, force=True),
136  topic=repr(topic), cols="*")
137  if exclude_cols:
138  exclude_cols = [x[0] if isinstance(x, (list, tuple)) else x for x in exclude_cols]
139  select_cols = [c for c in self._schema[typekey] if c not in exclude_cols]
140  sqlargs["cols"] = ", ".join(quote(c) for c in select_cols)
141  sql = self._get_dialect_option("view_template").strip().format(**sqlargs)
142 
143  return {"name": topic, "type": typename, "md5": typehash,
144  "sql": sql, "table_name": table_name, "view_name": view_name}

147  def _make_type_data(self, msg, extra_cols=(), rootmsg=None):
148  """
149  Returns full data dictionary for message type, including table name and SQL.
150 
151  @param rootmsg top message this message is nested under, if any
152  @param extra_cols additional table columns, as [(column name, column def)]
153  @return {"type": message type name as "pkg/Cls",
154  "table_name": message type table name,
155  "definition": message type definition,
156  "cols": [(column name, column type)],
157  "md5": message type definition MD5 hash, "sql": "CREATE TABLE .."}
158  """
159  rootmsg = rootmsg or msg
160  with rosapi.TypeMeta.make(msg, root=rootmsg) as m:
161  typename, typehash = (m.typename, m.typehash)
162 
163  cols = []
164  scalars = set(x for x in self._get_dialect_option("types") if x == rosapi.scalar(x))
165  for path, value, subtype in rosapi.iter_message_fields(msg, scalars=scalars):
166  coltype = self._make_column_type(subtype)
167  cols += [(".".join(path), coltype)]
168  cols.extend(extra_cols or [])
169  cols = list(zip(self._make_column_names([c for c, _ in cols]), [t for _, t in cols]))
170  namewidth = 2 + max(len(n) for n, _ in cols)
171  coldefs = ["%s %s" % (quote(n).ljust(namewidth), t) for n, t in cols]
172 
173  pkgname, clsname = typename.split("/", 1)
174  nameargs = {"type": typename, "hash": typehash, "package": pkgname, "class": clsname}
175  table_name = self._make_entity_name("table", nameargs)
176 
177  sqlargs = dict(nameargs, table=quote(table_name), cols="\n %s\n" % ",\n ".join(coldefs))
178  sql = self._get_dialect_option("table_template").strip().format(**sqlargs)
179  return {"type": typename, "md5": typehash,
180  "definition": rosapi.TypeMeta.make(msg).definition,
181  "table_name": table_name, "cols": cols, "sql": sql}

206  def _make_message_insert_sql(self, topic, msg, extra_cols=()):
207  """
208  Returns ("INSERT ..", [args]) for inserting into message type table.
209 
210  @param extra_cols list of additional table columns, as [(name, value)]
211  """
212  typekey = rosapi.TypeMeta.make(msg, topic).typekey
213  table_name = self._types[typekey]["table_name"]
214  sql, cols, args = self._sql_cache.get(table_name), [], []
215 
216  scalars = set(x for x in self._get_dialect_option("types") if x == rosapi.scalar(x))
217  for p, v, t in rosapi.iter_message_fields(msg, scalars=scalars):
218  if not sql: cols.append(".".join(p))
219  args.append(self._make_column_value(v, t))
220  args = tuple(args) + tuple(v for _, v in extra_cols)
221 
222  if not sql:
223  POSARG = self._get_dialect_option("posarg")
224  if extra_cols: cols.extend(c for c, _ in extra_cols)
225  sql = "INSERT INTO %s (%s) VALUES (%s)" % \
226  (quote(table_name), ", ".join(map(quote, cols)),
227  ", ".join([POSARG] * len(args)))
228  self._sql_cache[table_name] = sql
229 
230  return sql, args

233  def _make_update_sql(self, table, values, where=()):
234  """Returns ("UPDATE ..", [args])."""
235  POSARG = self._get_dialect_option("posarg")
236  sql, args, sets, filters = "UPDATE %s SET " % quote(table), [], [], []
237  for lst, vals in [(sets, values), (filters, where)]:
238  for k, v in vals.items() if isinstance(vals, dict) else vals:
239  lst.append("%s = %s" % (quote(k), POSARG))
240  args.append(self._make_column_value(v))
241  sql += ", ".join(sets) + (" WHERE " if filters else "") + " AND ".join(filters)
242  return sql, args

258  def _make_name(self, category, name, existing=()):
259  """
260  Returns a valid unique name for table/view/column.
261 
262  Replaces invalid characters and constrains length.
263  If name already exists, appends counter like " (2)".
264  """
265  MAXLEN_ARG = "maxlen_column" if "column" == category else "maxlen_entity"
266  MAXLEN = self._get_dialect_option(MAXLEN_ARG)
267  INVALID_RGX = self._get_dialect_option("invalid_char_regex")
268  INVALID_REPL = self._get_dialect_option("invalid_char_repl")
269  if not MAXLEN and not INVALID_RGX: return name
270 
271  name1 = re.sub(INVALID_RGX, INVALID_REPL, name) if INVALID_RGX else name
272  name2 = ellipsize(name1, MAXLEN)
273  counter = 2
274  while name2 in existing:
275  suffix = " (%s)" % counter
276  name2 = ellipsize(name1, MAXLEN - len(suffix)) + suffix
277  counter += 1
278  return name2

289  def _make_column_value(self, value, typename=None):
290  """Returns column value suitable for inserting to database."""
291  if not typename: return value
292 
293  v = value
294  if isinstance(v, (list, tuple)):
295  scalartype = rosapi.scalar(typename)
296  if scalartype in rosapi.ROS_TIME_TYPES:
297  v = [self._convert_time(x) for x in v]
298  elif scalartype not in rosapi.ROS_BUILTIN_TYPES:
299  if self._nesting: v = []
300  else: v = [rosapi.message_to_dict(x) for x in v]
301  else:
302  v = self._convert_column_value(v, typename)
303  elif rosapi.is_ros_time(v):
304  v = self._convert_time_value(v, typename)
305  elif typename not in rosapi.ROS_BUILTIN_TYPES:
306  v = json.dumps(rosapi.message_to_dict(v))
307  else:
308  v = self._convert_column_value(v, typename)
309  return v

312  def _make_column_type(self, typename, fallback=None):
313  """
314  Returns column type for SQL.
315 
316  @param fallback fallback typename to use for lookup if no mapping for typename
317  """
318  TYPES = self._get_dialect_option("types")
319  ARRAYTEMPLATE = self._get_dialect_option("arraytype_template")
320  DEFAULTTYPE = self._get_dialect_option("defaulttype")
321 
322  scalartype = rosapi.scalar(typename)
323  timetype = rosapi.get_ros_time_category(scalartype)
324  coltype = TYPES.get(typename)
325 
326  if not coltype and "[" not in typename and scalartype in TYPES:
327  coltype = TYPES[scalartype] # Bounded type like "string<=10"
328  if not coltype and scalartype in TYPES:
329  coltype = ARRAYTEMPLATE.format(type=TYPES[scalartype])
330  if not coltype and timetype in TYPES:
331  if typename != scalartype:
332  coltype = ARRAYTEMPLATE.format(type=TYPES[timetype])
333  else:
334  coltype = TYPES[timetype]
335  if not coltype and fallback:
336  coltype = self._make_column_type(fallback)
337  if not coltype:
338  coltype = DEFAULTTYPE or quote(typename)
339  return coltype

342  def _convert_column_value(self, value, typename):
343  """Returns ROS value converted to dialect value."""
344  ADAPTERS = self._get_dialect_option("adapters")
345  if not ADAPTERS: return value
346 
347  adapter, iterate = ADAPTERS.get(typename), False
348  if not adapter and isinstance(value, (list, tuple)):
349  adapter, iterate = ADAPTERS.get(rosapi.scalar(typename)), True
350  if adapter:
351  value = [adapter(x) for x in value] if iterate else adapter(value)
352  return value

355  def _convert_time_value(self, value, typename):
356  """Returns ROS time/duration value converted to dialect value."""
357  adapter = self._get_dialect_option("adapters").get(typename)
358  if adapter:
359  try: is_int = issubclass(adapter, int)
360  except Exception: is_int = False
361  v = rosapi.to_sec(value) if is_int else "%d.%09d" % rosapi.to_sec_nsec(value)
362  result = adapter(v)
363  else:
364  result = rosapi.to_decimal(value)
365  return result
